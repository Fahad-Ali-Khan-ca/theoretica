18 inline real approx_max_goldensection(RealFunction f, real a, real b) {
19
20 if(a > b) {
21 TH_MATH_ERROR("approx_max_goldensection", b, INVALID_ARGUMENT);
22 return nan();
23 }
24
25 real x1 = a;
26 real x2 = b;
27 real x3 = b - (b - a) / PHI;
28 real x4 = a + (b - a) / PHI;
29
30 unsigned int iter = 0;
31
32 while(abs(x2 - x1) > ROOT_APPROX_TOL && iter <= MAX_GOLDENSECTION_ITER) {
33
34 if(f(x3) > f(x4)) {
35 x2 = x4;
36 } else {
37 x1 = x3;
38 }
39
40 x3 = x2 - (x2 - x1) / PHI;
41 x4 = x1 + (x2 - x1) / PHI;
42
43 iter++;
44 }
45
46 if(iter > MAX_GOLDENSECTION_ITER) {
47 TH_MATH_ERROR("approx_max_goldensection", iter, NO_ALGO_CONVERGENCE);
48 return nan();
49 }
50
51 return (x2 + x1) / 2.0;
52 }
