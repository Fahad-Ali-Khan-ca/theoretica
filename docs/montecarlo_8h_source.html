41 template<unsigned int S>
42 inline real integral_crude(
43 real(*f)(vec<S>), vec<S, vec2> extremes,
44 PRNG& g, unsigned int N = 1000) {
45
46 real sum_y = 0;
47
48 // Sample the function at random points in the integration region
49 for (unsigned int i = 0; i < N; ++i) {
50
51 vec<S> v;
52 for (unsigned int k = 0; k < S; ++k)
53 v[k] = rand_uniform(extremes[k][0], extremes[k][1], g);
54
55 sum_y += f(v);
56 }
57
58 // Volume of the integration region
59 real volume = 1;
60 for (unsigned int i = 0; i < S; ++i)
61 volume *= (extremes[i][1] - extremes[i][0]);
62
63 return volume * sum_y / static_cast<real>(N);
64 }

126 template<unsigned int S>
127 inline real integral_quasi_crude(
128 real(*f)(vec<S>), vec<S, vec2> extremes,
129 unsigned int N = 1000, real alpha = 0) {
130
131 if(alpha == 0) {
132
133 // Find the only positive root of the polynomial
134 // x^s+1 - x - 1 = 0
135
136 alpha = 1.0 / root_bisection(
137 [](real x) {
138 return pow(x, S + 1) - x - 1;
139 }, 0, 2);
140 }
141
142 vec<S> v;
143 for (unsigned int i = 0; i < S; ++i)
144 v[i] = pow(alpha, i + 1);
145
146 return integral_quasi_crude(f, extremes, N, v);

158 inline real integral_hom(
159 real_function f,
160 real a, real b, real c, real d,
161 PRNG& g, unsigned int N = 1000) {
162
163 int N_inside = 0;
164
165 for (unsigned int i = 0; i < N; ++i) {
166
167 const real x_n = rand_uniform(a, b, g);
168 const real y_n = rand_uniform(c, d, g);
169
170 const real f_x = f(x_n);
171
172 if(f_x >= 0) {
173 if(f_x >= y_n)
174 N_inside++;
175 } else {
176 if(f_x < y_n)
177 N_inside--;
178 }
179 }
180
181 return (N_inside / static_cast<real>(N)) * (b - a) * (d - c);
182 }
