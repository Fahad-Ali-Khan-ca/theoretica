226 inline mat4 to_mat4() const {
227
228 real x = v.get(0);
229 real y = v.get(1);
230 real z = v.get(2);
231 real w = a;
232
233 mat4 res;
234
235 res.iat(0, 0) = 1 - (2 * square(y) + 2 * square(z));
236 res.iat(0, 1) = 2 * x * y + 2 * z * w;
237 res.iat(0, 2) = 2 * x * z - 2 * y * w;
238 res.iat(0, 3) = 0;
239
240 res.iat(1, 0) = 2 * x * y - 2 * z * w;
241 res.iat(1, 1) = 1 - (2 * square(x) + 2 * square(z));
242 res.iat(1, 2) = 2 * y * z + 2 * x * w;
243 res.iat(1, 3) = 0;
244
245 res.iat(2, 0) = 2 * x * z + 2 * y * w;
246 res.iat(2, 1) = 2 * y * z - 2 * x * w;
247 res.iat(2, 2) = 1 - (2 * square(x) + 2 * square(y));
248 res.iat(2, 3) = 0;
249
250 res.iat(3, 0) = 0;
251 res.iat(3, 1) = 0;
252 res.iat(3, 2) = 0;
253 res.iat(3, 3) = 1;
254
255 return res;
256 }

300 inline static vec3 rotate(const vec3& v, real rad, const vec3& axis) {
301
302 vec3 n_axis = axis.normalized();
303
304 quat q = quat(uroboro::cos(rad / 2.0),
305 n_axis * uroboro::sin(rad / 2.0));
306
307 quat q_inv = quat(uroboro::cos(rad / 2.0),
308 n_axis * (uroboro::sin(rad / 2.0) * -1));
309
310 quat p = quat(0, v);
311
312 return (q * p * q_inv).v;
313 }
