44 template<typename RealFunction>
45 inline real integral_midpoint(RealFunction f, real a, real b,
46 unsigned int steps = INTEGRATION_STEPS) {
47
48 if(steps == 0) {
49 TH_MATH_ERROR("integral_midpoint", steps, DIV_BY_ZERO);
50 return nan();
51 }
52
53 const real dx = (b - a) / steps;
54 real res = 0;
55
56 for (unsigned int i = 0; i < steps; ++i)
57 res += f(a + (i + 0.5) * dx);
58
59 return res * dx;
60 }

71 template<typename RealFunction>
72 inline real integral_trapezoid(RealFunction f, real a, real b,
73 unsigned int steps = INTEGRATION_STEPS) {
74
75 if(steps == 0) {
76 TH_MATH_ERROR("integral_trapezoid", steps, DIV_BY_ZERO);
77 return nan();
78 }
79
80 const real dx = (b - a) / steps;
81 real res = 0;
82
83 res += 0.5 * (f(a) + f(b));
84
85 for (unsigned int i = 1; i < steps; ++i)
86 res += f(a + i * dx);
87
88 return res * dx;
89 }

100 template<typename RealFunction>
101 inline real integral_simpson(RealFunction f, real a, real b,
102 unsigned int steps = INTEGRATION_STEPS) {
103
104 if(steps == 0) {
105 TH_MATH_ERROR("integral_simpson", steps, DIV_BY_ZERO);
106 return nan();
107 }
108
109 const real dx = (b - a) / (real) steps;
110 real res = 0;
111
112 // Sum terms by order of magnitude supposing that
113 // f stays at the same order inside the interval,
114 // to alleviate truncation errors
115
116 res += f(a) + f(b);
117
118 for (unsigned int i = 2; i < steps; i += 2)
119 res += 2 * f(a + i * dx);
120
121 for (unsigned int i = 1; i < steps; i += 2)
122 res += 4 * f(a + i * dx);
123
124 return res * dx / 3.0;
125 }

136 template<typename RealFunction>
137 inline real integral_romberg(
138 RealFunction f,
139 real a, real b,
140 unsigned int iter = 8) {
141
142 real T[iter][iter];
143
144 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
145
146 for (unsigned int j = 1; j < iter; ++j) {
147
148 // Composite Trapezoidal Rule
149 T[j][0] = integral_trapezoid(f, a, b, 1 << j);
150
151 // Richardson extrapolation
152 for (unsigned int k = 1; k <= j; ++k) {
153 const uint64_t coeff = 1 << (2 * k);
154 T[j][k] = (coeff * T[j][k - 1] - T[j - 1][k - 1]) / (coeff - 1);
155 }
156 }
157
158 // Return the best approximation
159 return T[iter - 1][iter - 1];
160 }

171 template<typename RealFunction>
172 inline real integral_romberg_tol(
173 RealFunction f,
174 real a, real b,
175 real tolerance = INTEGRATION_TOL) {
176
177 const unsigned int MAX_ROMBERG_ITER = 16;
178 real T[MAX_ROMBERG_ITER][MAX_ROMBERG_ITER];
179
180 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
181
182 for (unsigned int j = 1; j < 16; ++j) {
183
184 // Composite Trapezoidal Rule
185 T[j][0] = integral_trapezoid(f, a, b, 1 << j);
186
187 // Richardson extrapolation
188 for (unsigned int k = 1; k <= j; ++k) {
189 const uint64_t coeff = 1 << (2 * k);
190 T[j][k] = (coeff * T[j][k - 1] - T[j - 1][k - 1]) / (coeff - 1);
191 }
192
193 // Stop the algorithm when the desired precision has been reached
194 if(abs(T[j][j] - T[j - 1][j - 1]) < tolerance)
195 return T[j][j];
196 }
197
198 // Return the best approximation
199 return T[MAX_ROMBERG_ITER - 1][MAX_ROMBERG_ITER - 1];
200 }

275 template<typename RealFunction>
276 inline real integral_legendre(
277 RealFunction f, real a, real b, real* x, real* w, unsigned int n) {
278
279 const real mean = (b + a) / 2.0;
280 const real halfdiff = (b - a) / 2.0;
281
282 real res = 0;
283
284 for (int i = n - 1; i >= 0; --i)
285 res += w[i] * f(halfdiff * x[i] + mean);
286
287 return res * halfdiff;
288 }
