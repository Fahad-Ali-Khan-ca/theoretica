43 template<typename RealFunction>
44 inline real integral_midpoint(RealFunction f, real a, real b,
45 unsigned int steps = INTEGRATION_STEPS) {
46
47 if(steps == 0) {
48 TH_MATH_ERROR("integral_midpoint", steps, DIV_BY_ZERO);
49 return nan();
50 }
51
52 const real dx = (b - a) / steps;
53 real res = 0;
54
55 for (unsigned int i = 0; i < steps; ++i)
56 res += f(a + (i + 0.5) * dx);
57
58 return res * dx;
59 }

70 template<typename RealFunction>
71 inline real integral_trapezoid(RealFunction f, real a, real b,
72 unsigned int steps = INTEGRATION_STEPS) {
73
74 if(steps == 0) {
75 TH_MATH_ERROR("integral_trapezoid", steps, DIV_BY_ZERO);
76 return nan();
77 }
78
79 const real dx = (b - a) / steps;
80 real res = 0;
81
82 res += 0.5 * (f(a) + f(b));
83
84 for (unsigned int i = 1; i < steps; ++i)
85 res += f(a + i * dx);
86
87 return res * dx;
88 }

99 template<typename RealFunction>
100 inline real integral_simpson(RealFunction f, real a, real b,
101 unsigned int steps = INTEGRATION_STEPS) {
102
103 if(steps == 0) {
104 TH_MATH_ERROR("integral_simpson", steps, DIV_BY_ZERO);
105 return nan();
106 }
107
108 const real dx = (b - a) / (real) steps;
109 real res = 0;
110
111 // Sum terms by order of magnitude supposing that
112 // f stays at the same order inside the interval,
113 // to alleviate truncation errors
114
115 res += f(a) + f(b);
116
117 for (unsigned int i = 2; i < steps; i += 2)
118 res += 2 * f(a + i * dx);
119
120 for (unsigned int i = 1; i < steps; i += 2)
121 res += 4 * f(a + i * dx);
122
123 return res * dx / 3.0;
124 }

135 template<typename RealFunction>
136 inline real integral_romberg(
137 RealFunction f,
138 real a, real b,
139 unsigned int iter = 8) {
140
141 real T[iter][iter];
142
143 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
144
145 for (unsigned int j = 1; j < iter; ++j) {
146
147 // Composite Trapezoidal Rule
148 T[j][0] = integral_trapezoid(f, a, b, 1 << j);
149
150 // Richardson extrapolation
151 for (unsigned int k = 1; k <= j; ++k) {
152 const uint64_t coeff = 1 << (2 * k);
153 T[j][k] = (coeff * T[j][k - 1] - T[j - 1][k - 1]) / (coeff - 1);
154 }
155 }
156
157 // Return the best approximation
158 return T[iter - 1][iter - 1];
159 }

170 template<typename RealFunction>
171 inline real integral_romberg_tol(
172 RealFunction f,
173 real a, real b,
174 real tolerance = INTEGRATION_TOL) {
175
176 const unsigned int MAX_ROMBERG_ITER = 16;
177 real T[MAX_ROMBERG_ITER][MAX_ROMBERG_ITER];
178
179 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
180
181 for (unsigned int j = 1; j < 16; ++j) {
182
183 // Composite Trapezoidal Rule
184 T[j][0] = integral_trapezoid(f, a, b, 1 << j);
185
186 // Richardson extrapolation
187 for (unsigned int k = 1; k <= j; ++k) {
188 const uint64_t coeff = 1 << (2 * k);
189 T[j][k] = (coeff * T[j][k - 1] - T[j - 1][k - 1]) / (coeff - 1);
190 }
191
192 // Stop the algorithm when the desired precision has been reached
193 if(abs(T[j][j] - T[j - 1][j - 1]) < tolerance)
194 return T[j][j];
195 }
196
197 // Return the best approximation
198 return T[MAX_ROMBERG_ITER - 1][MAX_ROMBERG_ITER - 1];
199 }

210 template<typename RealFunction>
211 inline real integral_legendre(RealFunction f, real a, real b, const std::vector<real>& x) {
212
213 const std::vector<real> weights = legendre_weights(x);
214 const real mean = (b + a) / 2.0;
215 const real halfdiff = (b - a) / 2.0;
216
217 real res = 0;
218
219 for (int i = x.size() - 1; i >= 0; --i)
220 res += weights[i] * f(halfdiff * x[i] + mean);
221
222 return res * halfdiff;
223 }

280
281 const std::vector<real> weights = laguerre_weights(x);
282
283 const real exp_a = exp(-a);
284 const real exp_b = exp(-b);
285
286 real res = 0;
287
288 for (int i = x.size() - 1; i >= 0; --i)
289 res += weights[i] * (exp_a * f(x[i] + a) - exp_b * f(x[i] + b));
