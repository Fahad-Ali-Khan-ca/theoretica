66 real integral_simpson(real_function f, real a, real b,
67 unsigned int steps = INTEGRATION_STEPS) {
68
69 real dx = (b - a) / (real) steps;

89 real integral_romberg(real_function f, real a, real b, unsigned int order) {
90
91 if(order % 2 != 0) {
92 UMATH_ERROR("integral_romberg", order, IMPOSSIBLE_OPERATION);
93 return nan();
94 }
95
96 unsigned int iter = order / 2;
97
98 real T[iter][iter];
99
100 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
101
102 for (unsigned int j = 1; j < iter; ++j) {
103
104 // Composite Trapezoidal Rule
105 T[j][0] = integral_trapezoid(f, a, b, 1 << j);
106
107 // Richardson extrapolation
108 for (unsigned int k = 1; k <= j; ++k) {
109 T[j][k] =
110 T[j][k - 1]
111 + (T[j][k - 1] - T[j - 1][k - 1]) / ((1 << (2 * k)) - 1);
112 }
113 }
114
115 // Return the best approximation
116 return T[iter - 1][iter - 1];
117 }
