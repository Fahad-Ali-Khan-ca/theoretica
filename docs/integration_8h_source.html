42 template<typename RealFunction>
43 inline real integral_midpoint(RealFunction f, real a, real b,
44 unsigned int steps = INTEGRATION_STEPS) {
45
46 if(steps == 0) {
47 TH_MATH_ERROR("integral_midpoint", steps, DIV_BY_ZERO);
48 return nan();
49 }
50
51 real dx = (b - a) / steps;
52 real res = 0;
53
54 for (unsigned int i = 0; i < steps; ++i)
55 res += f(a + (i + 0.5) * dx);
56
57 return res * dx;
58 }

71 unsigned int steps = INTEGRATION_STEPS) {
72
73 if(steps == 0) {
74 TH_MATH_ERROR("integral_trapezoid", steps, DIV_BY_ZERO);
75 return nan();
76 }
77
78 real dx = (b - a) / steps;

80

82
83 for (unsigned int i = 1; i < steps; ++i)

100 template<typename RealFunction>
101 inline real integral_simpson(RealFunction f, real a, real b,
102 unsigned int steps = INTEGRATION_STEPS) {
103
104 if(steps == 0) {
105 TH_MATH_ERROR("integral_simpson", steps, DIV_BY_ZERO);
106 return nan();
107 }
108
109 real dx = (b - a) / (real) steps;
110 real res = 0;
111
112 res += f(a) + f(b);
113
114 for (unsigned int i = 1; i < steps; ++i) {
115
116 if(i % 2 == 0)
117 res += 2.0 * f(a + i * dx);
118 else
119 res += 4.0 * f(a + i * dx);
120
121 }
122
123 return res * dx / 3.0;
124 }

135 template<typename RealFunction>
136 inline real integral_romberg(
137 RealFunction f,
138 real a, real b,
139 unsigned int order = 16) {
140
141 if(order % 2 != 0) {
142 TH_MATH_ERROR("integral_romberg", order, IMPOSSIBLE_OPERATION);
143 return nan();
144 }
145
146 unsigned int iter = order / 2;
147
148 real T[iter][iter];
149
150 T[0][0] = (f(a) + f(b)) * (b - a) / 2.0;
151
152 for (unsigned int j = 1; j < iter; ++j) {
153
154 // Composite Trapezoidal Rule
155 T[j][0] = integral_trapezoid(f, a, b, 1 << j);
156
157 // Richardson extrapolation
158 for (unsigned int k = 1; k <= j; ++k) {
159 T[j][k] =
160 T[j][k - 1]
161 + (T[j][k - 1] - T[j - 1][k - 1]) / ((1 << (2 * k)) - 1);
162 }
163 }
164
165 // Return the best approximation
166 return T[iter - 1][iter - 1];
167 }
