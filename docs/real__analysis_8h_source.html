41 inline real sqrt(real x) {
42
43 if(x < 0) {
44 UMATH_ERROR("sqrt", x, OUT_OF_DOMAIN);
45 return nan();
46 }
47
48#ifdef UROBORO_X86
49
50 #ifdef MSVC_ASM
51 __asm {
52 fld x
53 fsqrt
54 }
55 #else
56 asm("fsqrt" : "+t"(x));
57 #endif
58
59 return x;
60#else
61
62 if(x < 1) {
63
64 if(x == 0)
65 return 0;
66
67 // Approximate sqrt(x) between 0 and 1
68 // The root of the inverse is the inverse of the root
69 // !!! Possible precision problems with smaller numbers
70 return 1.0 / sqrt(1.0 / x);
71 }
72
73 // Approximate sqrt(x) using Newton-Raphson
74 real y = x;
75 int i = 0;
76
77 while((square(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
78 y = (y + x / y) / 2.0;
79 i++;
80 }
81
82 return y;
83#endif

95 inline real cbrt(real x) {
96
97 if(x < 1) {
98
99 if(x == 0)
100 return 0;
101
102 // cbrt(x) is odd
103 if(x < 0)
104 return -cbrt(-x);
105
106 // Approximate cbrt between 0 and 1
107 // The root of the inverse is the inverse of the root
108 // !!! Possible precision problems with smaller numbers
109 return 1.0 / cbrt(1.0 / x);
110 }
111
112 // Approximate cbrt(x) using Newton-Raphson
113
114 real y = x;
115 int i = 0;
116
117 while((cube(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
118 y = (y * 2.0 + x / (y * y)) / 3.0;
119 i++;
120 }
121
122 return y;
123 }

344 template<typename T>
345 inline T pow(T x, int n) {
346
347 T res;
348 T x2 = x * x;
349 int i = 1;
350
351 if(n > 0) {
352
353 res = x;
354
355 for(; i < (n / 2); i += 2)
356 res *= x2;
357
358 for(; i < n; ++i)
359 res *= x;
360
361 } else if(n < 0) {
362
363 res = 1 / x;
364
365 for(; i < (n / 2); i += 2)
366 res /= x2;
367
368 for(; i < -n; ++i)
369 res /= x;
370 } else {
371 return 1.0;
372 }
373
374 return res;
375 }

382 inline real root(real x, int n) {
383
384 if(n % 2 == 0 && x < 0 || n == 0) {
385 UMATH_ERROR("root", n, OUT_OF_DOMAIN);
386 return nan();
387 }
388
389 if(n < 0)
390 return 1.0 / root(x, -n);
391
392 if(x < 1) {
393
394 if(x == 0)
395 return 0;
396
397 // Approximate root between 0 and 1
398 // The root of the inverse is the inverse of the root
399 // !!! Possible precision problems with smaller numbers
400 return 1.0 / root(1.0 / x, n);
401 }
402
403 // Approximate n-th root using Newton-Raphson
404
405 real y = x;
406 int i = 0;
407
408 while((pow(y, n) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
409 y = (y * (n - 1) + x / pow(y, n - 1)) / (real) n;
410 i++;
411 }
412
413 return y;
414 }

442 inline real exp(real x) {
443
444 // Domain reduction to [0, +inf]
445 if(x < 0)
446 return 1.0 / exp(abs(x));
447
448 real fract_x = fract(x);
449
450#ifdef UROBORO_X86
451
452 return pow(E, floor(x)) * exp_x86_norm(fract_x);
453
454#else
455
456 // Taylor series expansion
457 // Compute e^floor(x) * e^fract(x)
458
459 real res = 1;
460 real s_n = 1;
461
462 for (int i = 1; i < 9; ++i) {
463
464 // Recurrence formula to improve
465 // numerical stability and performance
466 s_n *= (fract_x / 2.0) / static_cast<real>(i);
467 res += s_n;
468 }
469
470 return pow(E, floor(x)) * res * res;
471
472#endif
473 }

504 asm("fsin" : "+t"(x));

509#else
510
511 // Clamp x between -2PI and 2PI
512 while(x >= 2 * PI)
513 x -= 2 * PI;
514
515 while(x <= -2 * PI)
516 x += 2 * PI;
517
518 real res = 0;
519
520 // Taylor series expansion
521 // sin(x) = sum( (-1)^i * x^(2i+1) / (2i+1)! )
522
523 for (int i = 0; i < TAYLOR_ORDER; ++i) {
524 res += (i % 2 == 0 ? 1 : -1)
525 * pow(x, 2 * i + 1)
526 / static_cast<real>(fact(2 * i + 1));
527 }

650 inline real atan(real x) {
651
652 // Domain reduction to [-1, 1]
653 if(abs(x) > 1.0)
654 return (PI2 - atan(1.0 / abs(x))) * sgn(x);
655
656 real x2 = x * x;
657
658 // Interpolating Chebyshev polynomial
659 // of order 9
660 return x * (0.999965
661 + x2 * (-0.331545
662 + x2 * (0.184464
663 + x2 * (-0.090752
664 + 0.023286 * x2))));
