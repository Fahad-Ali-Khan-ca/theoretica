49
50 if(x < 0) {

52 return nan();
53 }
54
55#ifdef UROBORO_X86
56

64 #endif
65
66 return x;
67#else
68
69 if(x < 1) {
70
71 if(x == 0)
72 return 0;
73
74 // Approximate sqrt(x) between 0 and 1
75 // The root of the inverse is the inverse of the root
76 // !!! Possible precision problems with smaller numbers
77 return 1.0 / sqrt(1.0 / x);

83

105 inline real cbrt(real x) {
106
107 if(x < 1) {
108
109 if(x == 0)
110 return 0;
111
112 // cbrt(x) is odd
113 if(x < 0)
114 return -cbrt(-x);
115
116 // Approximate cbrt between 0 and 1
117 // The root of the inverse is the inverse of the root
118 // !!! Possible precision problems with smaller numbers
119 return 1.0 / cbrt(1.0 / x);
120 }
121
122 // Approximate cbrt(x) using Newton-Raphson
123
124 real y = x;
125 int i = 0;
126
127 while((cube(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
128 y = (y * 2.0 + x / (y * y)) / 3.0;
129 i++;
130 }
131
132 return y;
133 }

413 template<typename T>
414 inline T pow(T x, int n) {
415
416 T res;
417 T x2 = x * x;
418 int i = 1;
419
420 if(n > 0) {
421
422 res = x;
423
424 for(; i < (n / 2); i += 2)
425 res *= x2;
426
427 for(; i < n; ++i)
428 res *= x;
429
430 } else if(n < 0) {
431
432 res = 1 / x;
433
434 for(; i < (n / 2); i += 2)
435 res /= x2;
436
437 for(; i < -n; ++i)
438 res /= x;
439 } else {
440 return static_cast<T>(1.0);
441 }
442
443 return res;
444 }

454 inline real root(real x, int n) {
455
456 if(n % 2 == 0 && x < 0 || n == 0) {
457 UMATH_ERROR("root", n, OUT_OF_DOMAIN);
458 return nan();
459 }
460
461 if(n < 0)
462 return 1.0 / root(x, -n);
463
464 if(x < 1) {
465
466 if(x == 0)
467 return 0;
468
469 // Approximate root between 0 and 1
470 // The root of the inverse is the inverse of the root
471 // !!! Possible precision problems with smaller numbers
472 return 1.0 / root(1.0 / x, n);
473 }
474
475 // Approximate n-th root using Newton-Raphson
476
477 real y = x;
478 int i = 0;
479
480 while((pow(y, n) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
481 y = (y * (n - 1) + x / pow(y, n - 1)) / (real) n;
482 i++;
483 }
484
485 return y;
486 }

521 inline real exp(real x) {
522
523 // Domain reduction to [0, +inf]
524 if(x < 0)
525 return 1.0 / exp(abs(x));
526
527 real fract_x = fract(x);
528
529#ifdef UROBORO_X86
530
531 return pow(E, floor(x)) * exp_x86_norm(fract_x);
532
533#else
534
535 // Taylor series expansion
536 // Compute e^floor(x) * e^fract(x)
537
538 real res = 1;
539 real s_n = 1;
540
541 for (int i = 1; i < 9; ++i) {
542
543 // Recurrence formula to improve
544 // numerical stability and performance
545 s_n *= (fract_x / 2.0) / static_cast<real>(i);
546 res += s_n;
547 }
548
549 return pow(E, floor(x)) * res * res;
550
551#endif
552 }

576 inline real sin(real x) {
577
578#ifdef UROBORO_X86
579
580 #ifdef MSVC_ASM
581 __asm {
582 fld x
583 fsin
584 }
585 #else
586 asm("fsin" : "+t"(x));
587 #endif
588
589 return x;
590
591#else
592
593 // Clamp x between -2PI and 2PI
594 while(x >= 2 * PI)
595 x -= 2 * PI;
596
597 while(x <= -2 * PI)
598 x += 2 * PI;
599
600 // Domain reduction to [-PI, PI]
601 if(x > PI) {
602 x = PI - x;
603 } else if(x < -PI) {
604 x = -PI - x;
605 }
606
607 real res = 0;
608
609 // Taylor series expansion
610 // sin(x) = sum( (-1)^i * x^(2i+1) / (2i+1)! )
611
612 for (int i = 0; i < TAYLOR_ORDER; ++i) {
613 res += (i % 2 == 0 ? 1 : -1)
614 * pow(x, 2 * i + 1)
615 / static_cast<real>(fact(2 * i + 1));
616 }
617
618 return res;
619#endif
620 }
