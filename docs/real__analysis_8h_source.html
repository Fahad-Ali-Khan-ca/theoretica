48 inline real sqrt(real x) {
49
50 if(x < 0) {
51 UMATH_ERROR("sqrt", x, OUT_OF_DOMAIN);
52 return nan();
53 }
54
55#ifdef THEORETICA_X86
56
57 #ifdef MSVC_ASM

59 fld x
60 fsqrt
61 }

63 asm("fsqrt" : "+t"(x));
64 #endif
65
66 return x;
67#else
68
69 if(x < 1) {
70
71 if(x == 0)
72 return 0;
73
74 // Approximate sqrt(x) between 0 and 1
75 // The root of the inverse is the inverse of the root
76 // !!! Possible precision problems with smaller numbers
77 return 1.0 / sqrt(1.0 / x);
78 }

81 real y = x;
82 int i = 0;
83

85 y = (y + x / y) / 2.0;
86 i++;
87 }
88

105 inline real cbrt(real x) {
106
107 if(x < 1) {
108
109 if(x == 0)
110 return 0;
111
112 // cbrt(x) is odd
113 if(x < 0)
114 return -cbrt(-x);
115
116 // Approximate cbrt between 0 and 1
117 // The root of the inverse is the inverse of the root
118 // !!! Possible precision problems with smaller numbers
119 return 1.0 / cbrt(1.0 / x);
120 }
121
122 // Approximate cbrt(x) using Newton-Raphson
123
124 real y = x;
125 int i = 0;
126
127 while((cube(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
128 y = (y * 2.0 + x / (y * y)) / 3.0;
129 i++;
130 }
131
132 return y;
133 }

329 inline real log2(real x) {
330
331 if(x <= 0) {
332
333 if(x == 0) {
334 UMATH_ERROR("log2", x, OUT_OF_RANGE);
335 return -inf();
336 }
337
338 UMATH_ERROR("log2", x, OUT_OF_DOMAIN);
339 return nan();
340 }
341
342#ifdef THEORETICA_X86
343
344 // Approximate the binary logarithm of x by
345 // exploiting x86 Assembly instructions
346 return fyl2x(x, 1.0);
347#else
348
349 // Domain reduction to [1, +inf)
350 if(x < 1)
351 return -log2(1 / x);
352
353 // Compute the biggest power of 2
354 // so that x <= 2^i
355 int i = 0;
356 while(x > (1 << i))
357 i++;
358
359 // Domain reduction to [1, 2]
360 x /= static_cast<real>(1 << i);
361
362 // Use a polynomial approximation of log2(x)
363 // in [1, 2] to compute the logarithm of the
364 // remainder.
365 // Note: Choosing a better interval or applying
366 // some transformations on the function a better
367 // approximation might be found.
368 real lr = 0;
369
370 // Exact powers of 2 don't need further computation
371 if(abs(x - 1) > MACH_EPSILON) {
372
373 lr = -3.23585 + x * (7.08614
374 + x * (-7.39388 + x * (5.6659
375 + x * (-2.90591 + x * (0.945908
376 + x * (-0.176734 + x * (0.0144404 * x)))))));
377 }
378
379 return static_cast<real>(i) + lr;
380#endif
381 }

450 template<typename T>
451 constexpr inline T pow(T x, int n) {
452
453 T res = T(0);
454 T x2 = x * x;
455 int i = 1;
456
457 if(n > 0) {
458
459 res = x;
460
461 for(; i < (n / 2); i += 2)
462 res *= x2;
463
464 for(; i < n; ++i)
465 res *= x;
466
467 } else if(n < 0) {
468
469 res = 1 / x;
470
471 for(; i < (n / 2); i += 2)
472 res /= x2;
473
474 for(; i < -n; ++i)
475 res /= x;
476 } else {
477 return static_cast<T>(1.0);
478 }
479
480 return res;
481 }

491 inline real root(real x, int n) {
492
493 if(n % 2 == 0 && x < 0 || n == 0) {
494 UMATH_ERROR("root", n, OUT_OF_DOMAIN);
495 return nan();
496 }
497
498 if(n < 0)
499 return 1.0 / root(x, -n);
500
501 if(x < 1) {
502
503 if(x == 0)
504 return 0;
505
506 // Approximate root between 0 and 1
507 // The root of the inverse is the inverse of the root
508 // !!! Possible precision problems with smaller numbers
509 return 1.0 / root(1.0 / x, n);
510 }
511
512 // Approximate n-th root using Newton-Raphson
513
514 real y = x;
515 int i = 0;
516
517 while((pow(y, n) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
518 y = (y * (n - 1) + x / pow(y, n - 1)) / (real) n;
519 i++;
520 }
521
522 return y;
523 }

559 inline real exp(real x) {
560
561 // Domain reduction to [0, +inf]
562 if(x < 0)
563 return 1.0 / exp(abs(x));
564
565 real fract_x = fract(x);
566
567#ifdef THEORETICA_X86
568
569 return pow(E, floor(x)) * exp_x86_norm(fract_x);
570
571#else
572
573 // Taylor series expansion
574 // Compute e^floor(x) * e^fract(x)
575
576 real res = 1;
577 real s_n = 1;
578
579 for (int i = 1; i < 9; ++i) {
580
581 // Recurrence formula to improve
582 // numerical stability and performance
583 s_n *= (fract_x / 2.0) / static_cast<real>(i);
584 res += s_n;
585 }
586
587 return pow(E, floor(x)) * res * res;
588
589#endif
590 }

614 inline real sin(real x) {
615
616#ifdef THEORETICA_X86
617
618 #ifdef MSVC_ASM
619 __asm {
620 fld x
621 fsin
622 }
623 #else
624 asm("fsin" : "+t"(x));
625 #endif
626
627 return x;
628
629#else
630
631 // Clamp x between -2PI and 2PI
632 while(x >= 2 * PI)
633 x -= 2 * PI;
634
635 while(x <= -2 * PI)
636 x += 2 * PI;
637
638 // Domain reduction to [-PI, PI]
639 if(x > PI) {
640 x = PI - x;
641 } else if(x < -PI) {
642 x = -PI - x;
643 }
644
645 real res = 0;
646
647 // Taylor series expansion
648 // sin(x) = sum( (-1)^i * x^(2i+1) / (2i+1)! )
649
650 for (int i = 0; i < TAYLOR_ORDER; ++i) {
651 res += (i % 2 == 0 ? 1 : -1)
652 * pow(x, 2 * i + 1)
653 / static_cast<real>(fact(2 * i + 1));
654 }
655
656 return res;
657#endif
658 }
