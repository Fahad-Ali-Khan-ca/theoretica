49
50 if(x < 0) {
51 UMATH_ERROR("sqrt", x, OUT_OF_DOMAIN);
52 return nan();
53 }
54
55#ifdef UROBORO_X86
56
57 #ifdef MSVC_ASM

60 fsqrt

64 #endif
65
66 return x;
67#else
68
69 if(x < 1) {
70
71 if(x == 0)
72 return 0;
73
74 // Approximate sqrt(x) between 0 and 1
75 // The root of the inverse is the inverse of the root
76 // !!! Possible precision problems with smaller numbers
77 return 1.0 / sqrt(1.0 / x);
78 }

80 // Approximate sqrt(x) using Newton-Raphson
81 real y = x;
82 int i = 0;
83
84 while((square(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
85 y = (y + x / y) / 2.0;
86 i++;
87 }

105 inline real cbrt(real x) {
106
107 if(x < 1) {
108
109 if(x == 0)
110 return 0;
111
112 // cbrt(x) is odd
113 if(x < 0)
114 return -cbrt(-x);
115
116 // Approximate cbrt between 0 and 1
117 // The root of the inverse is the inverse of the root
118 // !!! Possible precision problems with smaller numbers
119 return 1.0 / cbrt(1.0 / x);
120 }
121
122 // Approximate cbrt(x) using Newton-Raphson
123
124 real y = x;
125 int i = 0;
126
127 while((cube(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
128 y = (y * 2.0 + x / (y * y)) / 3.0;
129 i++;
130 }
131
132 return y;
133 }

396 inline T pow(T x, int n) {
397
398 T res;
399 T x2 = x * x;
400 int i = 1;
401
402 if(n > 0) {
403
404 res = x;
405
406 for(; i < (n / 2); i += 2)
407 res *= x2;
408
409 for(; i < n; ++i)
410 res *= x;
411
412 } else if(n < 0) {
413
414 res = 1 / x;
415
416 for(; i < (n / 2); i += 2)
417 res /= x2;
418
419 for(; i < -n; ++i)
420 res /= x;
421 } else {
422 return static_cast<T>(1.0);
423 }
424
425 return res;
426 }

436 inline real root(real x, int n) {
437
438 if(n % 2 == 0 && x < 0 || n == 0) {
439 UMATH_ERROR("root", n, OUT_OF_DOMAIN);
440 return nan();
441 }
442
443 if(n < 0)
444 return 1.0 / root(x, -n);
445
446 if(x < 1) {
447
448 if(x == 0)
449 return 0;
450
451 // Approximate root between 0 and 1
452 // The root of the inverse is the inverse of the root
453 // !!! Possible precision problems with smaller numbers
454 return 1.0 / root(1.0 / x, n);
455 }
456
457 // Approximate n-th root using Newton-Raphson
458
459 real y = x;
460 int i = 0;
461
462 while((pow(y, n) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
463 y = (y * (n - 1) + x / pow(y, n - 1)) / (real) n;
464 i++;
465 }
466
467 return y;
468 }

503 inline real exp(real x) {
504
505 // Domain reduction to [0, +inf]
506 if(x < 0)
507 return 1.0 / exp(abs(x));
508
509 real fract_x = fract(x);
510
511#ifdef UROBORO_X86
512
513 return pow(E, floor(x)) * exp_x86_norm(fract_x);
514
515#else
516
517 // Taylor series expansion
518 // Compute e^floor(x) * e^fract(x)
519
520 real res = 1;
521 real s_n = 1;
522
523 for (int i = 1; i < 9; ++i) {
524
525 // Recurrence formula to improve
526 // numerical stability and performance
527 s_n *= (fract_x / 2.0) / static_cast<real>(i);
528 res += s_n;
529 }
530
531 return pow(E, floor(x)) * res * res;
532
533#endif
534 }

558 inline real sin(real x) {
559
560#ifdef UROBORO_X86
561
562 #ifdef MSVC_ASM
563 __asm {
564 fld x
565 fsin
566 }
567 #else
568 asm("fsin" : "+t"(x));
569 #endif
570
571 return x;
572
573#else
574
575 // Clamp x between -2PI and 2PI
576 while(x >= 2 * PI)
577 x -= 2 * PI;
578
579 while(x <= -2 * PI)
580 x += 2 * PI;
581
582 // Domain reduction to [-PI, PI]
583 if(x > PI) {
584 x = PI - x;
585 } else if(x < -PI) {
586 x = -PI - x;
587 }
588
589 real res = 0;
590
591 // Taylor series expansion
592 // sin(x) = sum( (-1)^i * x^(2i+1) / (2i+1)! )
593
594 for (int i = 0; i < TAYLOR_ORDER; ++i) {
595 res += (i % 2 == 0 ? 1 : -1)
596 * pow(x, 2 * i + 1)
597 / static_cast<real>(fact(2 * i + 1));
598 }
599
600 return res;
601#endif
602 }
