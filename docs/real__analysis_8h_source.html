52 template<typename UnsignedIntType = uint64_t>
53 inline UnsignedIntType isqrt(UnsignedIntType n) {
54
55 // Upper bound
56 UnsignedIntType upper = n + 1;
57
58 // Lower bound
59 UnsignedIntType lower = 0;
60
61 // Carry for safe long division
62 UnsignedIntType c = 0;
63
64 while(lower != upper - 1) {
65
66 // Compute carry for long division by 2
67 c = ((lower % 2 != 0) && (upper % 2 != 0)) ? 1 : 0;
68
69 // Safer division by 2 for big numbers
70 const UnsignedIntType m = (lower >> 1) + (upper >> 1) + c;
71
72 // Using division instead of multiplication avoids
73 // overflows which would remove significant bits
74 const UnsignedIntType q = n / m;
75
76 if(m > q)
77 upper = m;
78 else if(m < q)
79 lower = m;
80 else
81 return m;
82 }
83
84 return lower;
85 }

197 inline real cbrt(real x) {
198
199 if(x < 1) {
200
201 if(x == 0)
202 return 0;
203
204 // cbrt(x) is odd
205 if(x < 0)
206 return -cbrt(-x);
207
208 // Approximate cbrt between 0 and 1
209 // The root of the inverse is the inverse of the root
210 // !!! Possible precision problems with smaller numbers
211 return 1.0 / cbrt(1.0 / x);
212 }
213
214 // Approximate cbrt(x) using Newton-Raphson
215 real y = x;
216 unsigned int i = 0;
217
218 while((cube(y) - x) > ROOT_APPROX_TOL && i < MAX_NEWTON_ITER) {
219 y = (y * 2.0 + x / (y * y)) / 3.0;
220 i++;
221 }
222
223 return y;

420 inline real log2(real x) {
421
422 if(x <= 0) {
423
424 if(x == 0) {
425 TH_MATH_ERROR("log2", x, OUT_OF_RANGE);
426 return -inf();
427 }
428
429 TH_MATH_ERROR("log2", x, OUT_OF_DOMAIN);
430 return nan();
431 }
432
433#ifdef THEORETICA_X86
434
435 // Approximate the binary logarithm of x by
436 // exploiting x86 Assembly instructions
437 return fyl2x(x, 1.0);
438#else
439
440 // Domain reduction to [1, +inf)
441 if(x < 1)
442 return -log2(1 / x);
443
444 // Compute the biggest power of 2
445 // so that x <= 2^i
446 int i = 0;
447 while(x > (1 << i))
448 i++;
449
450 // Domain reduction to [1, 2]
451 x /= (1 << i);
452
453 // Use the Taylor expansion of the logarithm
454 // ln(1 + x) = \sum_k^n (-1)^(k+1) x^k / k
455 real log_z = 0;
456 const real z = x - 1;
457
458 // Exact powers of 2 don't need further computation
459 if(abs(z) > MACH_EPSILON) {
460
461 int sign = 1;
462 real pow_z = z;
463 log_z = z;
464
465 for (int i = 2; i <= 24; ++i) {
466
467 pow_z *= z;
468 sign *= -1;
469
470 log_z += sign * pow_z / i;
471 }
472 }
473
474 return i + (log_z / LN2);
475#endif
476 }

572 template<typename UnsignedIntType = uint64_t>
573 inline UnsignedIntType pad2(UnsignedIntType x) {
574
575 UnsignedIntType bit = 0;
576
577 for (int i = (sizeof(UnsignedIntType) * 8 - 1); i > 0; --i) {
578 if(x & ((UnsignedIntType) 1 << i)) {
579
580 // Find the highest set bit
581 bit = i;
582
583 // Check if x is a power of 2
584 for (int j = 0; j < i; ++j)
585 if(x & ((UnsignedIntType) 1 << j))
586 return (1 << (bit + 1));
587 }
588 }
589
590 return (1 << bit);

598 template<typename T = real>
599 TH_CONSTEXPR inline T pow(T x, int n) {
600
601 if(n > 0) {
602
603 T res = x;
604 T x_sqr = x * x;
605 int i = 1;
606
607 // Self-multiply up to biggest power of 2
608 while ((i * 2) <= n) {
609 res = res * res;
610 i *= 2;
611 }
612
613 // Multiply by x^2 for remaining even power
614 for (; i < (n - 1); i += 2)
615 res = res * x_sqr;
616
617 // Multiply for remaining powers
618 for (; i < n; ++i)
619 res = res * x;
620
621 return res;
622
623 } else if(n < 0) {
624 return 1.0 / pow(x, -n);
625 } else {
626 return 1;
627 }
628 }

712 inline real exp(real x) {
713
714 // Domain reduction to [0, +inf]
715 if(x < 0)
716 return 1.0 / exp(-x);
717
718 const real fract_x = fract(x);
719 const int floor_x = floor(x);
720
721 // Taylor series expansion
722 // Compute e^floor(x) * e^fract(x)
723
724 real res = 1;
725 real s_n = 1;
726
727 for (int i = 1; i <= TAYLOR_ORDER; ++i) {
728
729 // Recurrence formula to improve
730 // numerical stability and performance
731 s_n *= fract_x / (i * 4);
732 res += s_n;
733 }
734
735 // The fractional part is divided by 4 to improve convergence
736 const real sqr_r = res * res;
737 return pow(E, floor_x) * sqr_r * sqr_r;
738 }

763 inline real root(real x, int n) {
764
765 if(((n % 2 == 0) && (x < 0)) || (n == 0)) {
766 TH_MATH_ERROR("root", n, OUT_OF_DOMAIN);
767 return nan();
768 }
769
770 if(n < 0)
771 return 1.0 / root(x, -n);
772
773 // Trivial cases
774 if(n == 1)
775 return x;
776
777 if(n == 2)
778 return sqrt(x);
779
780 if(n == 3)
781 return cbrt(x);
782
783 if(x < 1) {
784
785 if(x == 0)
786 return 0;
787
788 // Approximate root between 0 and 1
789 // The root of the inverse is the inverse of the root
790 // !!! Possible precision problems with smaller numbers
791 return 1.0 / root(1.0 / x, n);
792 }
793
794 // Approximate n-th root using Newton-Raphson
795 real y = x;
796 unsigned int i = 0;
797
798 while(i < MAX_NEWTON_ITER) {
799
800 const real y_pow = pow(y, n - 1);
801
802 if((y_pow * y - x) < ROOT_APPROX_TOL)
803 break;
804
805 y = (y * (n - 1) + x / y_pow) / (real) n;
806 i++;
807 }
808
809 if(i >= MAX_NEWTON_ITER) {
810 TH_MATH_ERROR("root", i, NO_ALGO_CONVERGENCE);
811 return nan();
812 }
813
814 return y;
815 }

842 if(abs(x) >= TAU)

845 // Domain reduction to [-PI, PI]
846 if(x > PI)
847 x = PI - x;
848 else if(x < -PI)

852 real res = x;
853 real s = x;
