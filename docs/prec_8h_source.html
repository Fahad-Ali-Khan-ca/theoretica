192 inline estimate_result compute_estimate(
193 std::string name,
194 RealFunction fApprox,
195 RealFunction fExp,
196 interval k,
197 Real tolerance = state.defaultTolerance,
198 bool quiet = false,
199 unsigned int n = state.defaultIterations,
200 FailFunction fail = state.defaultFailFunction) {
201
202 estimate_result result;
203
204 Real sum = 0;
205 Real sum_sqr = 0;
206 Real sum_abs = 0;
207 Real max = 0;
208
209 Real measure = k.length();
210 Real dx = measure / n;
211 Real x, coeff;
212
213 Real diff = std::abs(fApprox(k.a) - fExp(k.a));
214
215 sum += diff;
216 sum_sqr += diff * diff;
217 sum_abs += std::abs(fExp(k.a));
218 max = diff;
219
220 for (unsigned int i = 1; i < n; ++i) {
221
222 x = k.a + i * dx;
223 diff = std::abs(fApprox(x) - fExp(x));
224
225 if(diff > max)
226 max = diff;
227
228 if(i % 2 == 0)
229 coeff = 2;
230 else
231 coeff = 4;
232
233 sum += coeff * diff;
234 sum_sqr += coeff * diff * diff;
235 sum_abs += coeff * fExp(x);
236 }
237
238 diff = std::abs(fApprox(k.b) - fExp(k.b));
239
240 sum += diff;
241 sum_sqr += diff * diff;
242 sum_abs += std::abs(fExp(k.b));
243
244 if(diff > max)
245 max = diff;
246
247 result.funcName = name;
248 result.quiet = quiet;
249 result.k = k;
250 result.iterations = n;
251 result.abs_err = sum;
252 result.max_err = max;
253 result.mean_err = (sum * dx / 3.0) / measure;
254 result.rms_err = std::sqrt((sum_sqr * dx / 3.0) / measure);
255 result.rel_err = (sum * dx / 3.0) / (sum_abs * dx / 3.0);
256 result.tolerance = tolerance;
257
258 result.failed = fail(result);
259
260 state.estimationResults[result.funcName].push_back(result);
261 state.totalTests++;
262
263 return result;
264 }

326 inline equation_result eval_equation(
327 const std::string& name,
328 Real evaluated, Real expected,
329 Real tolerance = CHEBYSHEV_TOLERANCE, bool quiet = false) {
330
331 equation_result eq;
332 Real diff = std::abs(expected - evaluated);
333
334 if(diff > tolerance) {
335 state.failedTests++;
336 eq.failed = true;
337 } else {
338 eq.failed = false;
339 }
340
341 state.totalTests++;
342
343 eq.funcName = name;
344 eq.diff = diff;
345 eq.expected = expected;
346 eq.evaluated = evaluated;
347 eq.tolerance = tolerance;
348 eq.quiet = quiet;
349
350 state.equationResults[eq.funcName].push_back(eq);
351 return eq;
352 }

445 inline void run() {
446
447 if(state.estimationRequests.size() + state.estimationCustomRequests.size()) {
448
449 if(!state.quiet) {
450 std::cout << "\n" << std::left << std::setw(20) << "Function" << " | "
451 << std::setw(12) << "Int. Min." << " | "
452 << std::setw(12) << "Int. Max." << " | "
453 << std::setw(12) << "Mean Err." << " | "
454 << std::setw(12) << "RMS Err." << " | "
455 << std::setw(12)<< "Max Err." << " | "
456 << std::setw(12)<< "Rel. Err." << std::endl;
457 }
458
459 if(state.outputToFile)
460 state.outputFile << "Function, Int. Min., Int. Max., Mean Err., "
461 << "RMS Err., Max Err., Rel. Err." << std::endl;
462
463 for(const auto& r : state.estimationRequests) {
464
465 if(!state.pickedTests.empty() && !state.pickedTests[r.funcName])
466 continue;
467
468 auto res = compute_estimate(r);
469
470 for(size_t i = 0; i < res.size(); i++) {
471
472 if(res[i].failed)
473 state.failedTests++;
474
475 // Skip test if only picked tests are to be executed
476 if(state.estimateOnlyFailed && !res[i].failed)
477 continue;
478
479 if(!state.quiet)
480 print_estimate(res[i], i);
481
482 if(state.outputToFile) {
483 state.outputFile << res[i].funcName << ", " << res[i].k.a << ", " << res[i].k.b << ", "
484 << res[i].mean_err << ", " << res[i].rms_err << ", "
485 << res[i].max_err << ", " << res[i].rel_err << std::endl;
486 }
487
488 }
489
490 }
491 state.estimationRequests.clear();
492
493 for(const auto& r : state.estimationCustomRequests) {
494
495 // Skip test if only picked tests are to be executed
496 if(!state.pickedTests.empty() && !state.pickedTests[r.funcName])
497 continue;
498
499 std::vector<estimate_result> res;
500
501 for (size_t j = 0; j < r.intervals.size(); ++j) {
502
503 res.push_back(r.f(r.intervals[j], r.tolerance, r.iterations));
504 res[j].funcName = r.funcName;
505 res[j].k = r.intervals[j];
506 res[j].tolerance = r.tolerance;
507 res[j].iterations = r.iterations;
508 res[j].quiet = r.quiet;
509
510 state.totalTests++;
511
512 if(res[j].failed)
513 state.failedTests++;
514 }
515
516 for(size_t i = 0; i < res.size(); i++) {
517
518 if(state.estimateOnlyFailed && !res[i].failed)
519 continue;
520
521 if(!state.quiet)
522 print_estimate(res[i], i);
523
524 if(state.outputToFile) {
525 state.outputFile << res[i].funcName << ", " << res[i].k.a << ", " << res[i].k.b << ", "
526 << res[i].mean_err << ", " << res[i].rms_err << ", "
527 << res[i].max_err << ", " << res[i].rel_err << std::endl;
528 }
529
530 }
531
532 }
533 state.estimationCustomRequests.clear();
534 }
535
536
537 if(state.equationRequests.size()) {
538
539 if(!state.quiet) {
540 std::cout << "\n" << std::setw(20) << "Function" << " | "
541 << std::setw(12) << "Eval. Value" << " | "
542 << std::setw(12) << "Exp. Value" << " | "
543 << std::setw(12) << "Diff." << " | "
544 << std::setw(12) << "Tol." << std::endl;
545 }
546
547 if(state.outputToFile)
548 state.outputFile << "Function, Eval. Value, Exp. Value, Diff., Tol." << std::endl;
549
550 for (size_t i = 0; i < state.equationRequests.size(); i++) {
551
552 // Skip test if only picked tests are to be executed
553 if( !state.pickedTests.empty() &&
554 !state.pickedTests[state.equationRequests[i].funcName])
555 continue;
556
557 equation_result res = eval_equation(state.equationRequests[i]);
558
559 if(state.equalsOnlyFailed && !res.failed)
560 return;
561
562 if(!state.quiet)
563 print_equation(res, i);
564
565 if(state.outputToFile) {
566 state.outputFile << res.funcName << ", " << res.evaluated << ", " << res.expected
567 << ", " << res.diff << ", " << res.tolerance << std::endl;
568 }
569 }
570
571 state.equationRequests.clear();
572 }
573 }

577 inline void terminate(bool exit = true) {
578
579 if(state.estimationRequests.size() +
580 state.equationRequests.size() +
581 state.estimationCustomRequests.size())
582 run();
583
584 std::cout << "\nFinished testing " << state.moduleName << std::endl;
585 std::cout << state.totalTests << " total tests, " << state.failedTests << " failed (" <<
586 (state.failedTests / (double) state.totalTests) * 100 << "%)" << std::endl;
587
588 std::cout << "Results have been saved in "
589 << state.outputFolder << "prec_" << state.moduleName << ".csv" << std::endl;
590 state.outputFile.close();
591
592 state = prec_state();
593
594 if(exit)
595 std::exit(state.failedTests);
596 }
