27 inline polynomial<real> gen_polyn_recurr(
28 polynomial<real> P0,
29 polynomial<real> P1,
30 polyn_recurr_formula f,
31 unsigned int n) {
32
33 if(n == 0)
34 return P0;
35
36 if(n == 1)
37 return P1;
38
39 polynomial<real> Pl;
40 for (unsigned int l = 2; l <= n; ++l) {
41
42 Pl = f(P0, P1, l);
43
44 P0 = P1;
45 P1 = Pl;
46 }
47
48 return Pl;
49 }

56 inline polynomial<real> legendre_polyn_recurr(
57 polynomial<real> P0, polynomial<real> P1, unsigned int l) {
58
59 return ((2 * l - 1) * P1 * polynomial<real>({0, 1}) - (l - 1) * P0) / l;

168 inline std::vector<real> legendre_roots(unsigned int n) {
169
170 if(n == 0)
171 return {};
172
173 if(n == 1)
174 return {0};
175
176 polynomial<real> P = legendre_polynomial(n);
177 std::vector<real> roots;
178 roots.reserve(n);
179
180 for (unsigned int i = 1; i <= n; ++i) {
181 roots.push_back(
182 approx_polyn_root_newton(P, (2.0 / (n + 1)) * i - 1.0)
183 );
184 }
185
186 return roots;
187 }

193 inline std::vector<real> legendre_weights(const std::vector<real>& roots) {
194
195 const unsigned int n = roots.size();
196 const polynomial<real> dP = deriv_polynomial(legendre_polynomial(n));
197
198 std::vector<real> weights;
199 weights.reserve(n);
200
201 for (unsigned int i = 0; i < n; ++i) {
202 weights.push_back(
203 2.0 / ((1 - square(roots[i])) * square(dP(roots[i])))
204 );
205 }
