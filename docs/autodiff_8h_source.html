66 template<unsigned int N>
67 inline vec<real, N> gradient_mono(dual(*f)(vec<dual, N>), const vec<real, N>& x) {
68
69 vec<real, N> res;
70 res.resize(x.size());
71
72 vec<dual, N> dual_x;
73 dual_x.resize(x.size());
74
75 for (unsigned int i = 0; i < x.size(); ++i)
76 dual_x[i].a = x[i];
77
78 // Re-evaluate the function with varying
79 // dual part to extract each derivative
80 for (unsigned int i = 0; i < x.size(); ++i) {
81 dual_x[i].b = 1;
82 res[i] = f(dual_x).Dual();
83 dual_x[i].b = 0;
84 }
85
86 return res;
87 }

151 template<unsigned int N, unsigned int M>
152 inline mat<real, M, N> jacobian(
153 vec<d_real<N>, M>(*f)(d_vec<N>), const vec<real, N>& x) {
154
155 vec<d_real<N>, M> res = f(d_real<N>::make_argument(x));
156
157 // Construct the jacobian matrix
158 mat<real, M, N> J;
159 J.resize(res.size(), x.size());
160
161 for (unsigned int j = 0; j < J.rows(); ++j)
162 for (unsigned int i = 0; i < res[j].v.size(); ++i)
163 J(j, i) = res[j].v[i];
164
165 return J;
166 }

184 template<unsigned int N>
185 inline vec<real, N> curl(d_vec<N>(*f)(d_vec<N>), const vec<real, N>& x) {
186
187 if(x.size() != 3) {
188 TH_MATH_ERROR("th::curl", x.size(), INVALID_ARGUMENT);
189 return vec<real, N>(nan(), x.size());
190 }
191
192 mat<real, N, N> J = jacobian<N, N>(f, x);
193 J.resize(3, 3);
194
195 vec<real, N> res;
196 res.resize(3);
197
198 res[0] = J(2, 1) - J(1, 2);
199 res[1] = J(0, 2) - J(2, 0);
200 res[2] = J(1, 0) - J(0, 1);
201
202 return res;
203 }

259
260 real res = 0;

262 d.resize(x.size());
263
264 for (unsigned int i = 0; i < x.size(); ++i)
265 d[i].a = x[i];
266
