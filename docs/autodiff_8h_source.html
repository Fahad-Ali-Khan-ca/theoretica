132 template<unsigned int N, unsigned int M>
133 inline mat<M, N> jacobian(vec<M, multidual<N>>(*f)(vec<N, multidual<N>>), const vec<N, real>& x) {
134
135 vec<M, multidual<N>> res = f(multidual<N>::make_argument(x));
136
137 // Construct the jacobian matrix
138 mat<M, N> J;
139 for (unsigned int i = 0; i < N; ++i) {
140 for (unsigned int j = 0; j < M; ++j) {
141 J.iat(j, i) = res.at(j).Dual().at(i);
142 }
143 }
144
145 return J;
146 }

163 inline vec<3> curl(vec<3, multidual<3>>(*f)(vec<3, multidual<3>>), const vec3& x) {
164
165 mat3 J = jacobian<3, 3>(f, x);
166 vec3 res;
167
168 res.at(0) = J.iat(2, 1) - J.iat(1, 2);
169 res.at(1) = J.iat(0, 2) - J.iat(2, 0);
170 res.at(2) = J.iat(1, 0) - J.iat(0, 1);
171
172 return res;
173 }
