102 template<unsigned int N, unsigned int M>
103 inline mat<M, N> jacobian(vec<M, multidual<N>>(*f)(vec<N, multidual<N>>), const vec<N, real>& x) {
104
105 vec<M, multidual<N>> res = f(multidual<N>::pack_function_arg(x));
106
107 // Construct the jacobian matrix
108 mat<M, N> J;
109 for (int i = 0; i < N; ++i) {
110 for (int j = 0; j < M; ++j) {
111 J.iat(j, i) = res.at(j).Dual().at(i);
112 }
113 }
114
115 return J;
116 }

122 inline vec<3> curl(vec<3, multidual<3>>(*f)(vec<3, multidual<3>>), const vec<3, real>& x) {
123
124 mat3 J = jacobian<3, 3>(f, x);
125 vec3 res;
126
127 res.at(0) = J.iat(2, 1) - J.iat(1, 2);
128 res.at(1) = J.iat(0, 2) - J.iat(2, 0);
129 res.at(2) = J.iat(1, 0) - J.iat(0, 1);
130
131 return res;
