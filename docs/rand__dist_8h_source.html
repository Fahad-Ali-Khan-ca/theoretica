27 inline real rand_uniform(real a, real b, PRNG& g, uint64_t prec = RAND_REAL_PREC) {
28
29 // Generate a uniform random real number in [0, 1]

87 inline real rand_gaussian_polar(real mean, real sigma, PRNG& g) {
88
89 static real spare;
90 static bool has_spare = false;
91
92 if(has_spare) {
93 has_spare = false;
94 return mean + spare * sigma;
95 }
96
97 real x, y, s;
98
99 // Generate a random point inside the unit circle
100 do {
101
102 x = rand_uniform(-1, 1, g);
103 y = rand_uniform(-1, 1, g);
104 s = square(x) + square(y);
105
106 } while(s >= 1 || s <= MACH_EPSILON);
107
108 // Project the point
109 s = sqrt(-2 * ln(s) / s);
110
111 // Keep the second generated value for future calls
112 spare = y * s;
113 has_spare = true;
114
115 return mean + sigma * x * s;
116 }

124 inline real rand_gaussian_boxmuller(real mean, real sigma, PRNG& g) {
125
126 static real spare;
127 static bool has_spare = false;
128
129 if(has_spare) {
130 has_spare = false;
131 return mean + spare * sigma;
132 }
133
134 // Generate a random point inside the unit circle
135
136 const real x = rand_uniform(0, 1, g);
137 const real y = rand_uniform(0, 1, g);
138
139 const real x_transf = sqrt(-2 * ln(x));
140
141 const real u = x_transf * cos(TAU * y);
142 const real v = x_transf * sin(TAU * y);
143
144 spare = v;
145 has_spare = true;
146
147 return mean + sigma * u;
148 }

165 inline real rand_gaussian_clt(
166 real mean, real sigma,
167 PRNG& g, unsigned int N = 10) {
168
169 real s = 0;
170 for (unsigned int i = 0; i < N; ++i)
171 s += rand_uniform(-1, 1, g);
172
173 // f(u) = 1/2 (in [-1, 1])
174 // E[u] = 0
175 // V[u] = 1 / sqrt(3N)
176
177 return mean + (s / N) * sigma * sqrt(3 * N);
178 }

190 inline real rand_exponential(real lambda, PRNG& g) {
191
