65 inline real rand_gaussian_polar(real mean, real sigma, PRNG& g) {
66
67 static real spare;
68 static double has_spare = false;
69
70 if(has_spare) {
71 has_spare = false;
72 return spare;
73 }
74
75 real x, y, s;
76
77 // Generate a random point inside the unit circle
78 do {
79
80 x = rand_real(-1, 1, g);
81 y = rand_real(-1, 1, g);
82 s = square(x) + square(y);
83
84 } while(s >= 1 || s <= MACH_EPSILON);
85
86 // Project the point onto the unit circumference
87 s = sqrt(-2 * ln(s) / s);
88
89 spare = y;
90 has_spare = true;
91
92 return mean + sigma * x * s;
93 }
