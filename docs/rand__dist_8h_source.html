33 inline real rand_uniform(real a, real b, PRNG& g, uint64_t prec = RAND_REAL_PREC) {
34
35 // Generate a uniform random real number in [0, 1]
36 real x = (g() % prec) / static_cast<real>(prec);
37
38 // Transform to target interval
39 return a + (b - a) * x;
40 }

78 inline real rand_trycatch(stat_function f,
79 const vec_buff& theta,
80 real x1, real x2,
81 real y1, real y2, PRNG& g,
82 unsigned int max_iter = MAX_TRYANDCATCH_ITER) {
83
84 real x;
85 real y;
86
87 unsigned int iter = 0;
88
89 do {
90 x = rand_uniform(x1, x2, g);
91 y = rand_uniform(y1, y2, g);
92 iter++;
93 } while(y > f(x, theta) && iter <= max_iter);
94
95 if(iter > max_iter) {
96 TH_MATH_ERROR("rand_dist_tac", iter, NO_ALGO_CONVERGENCE);
97 return nan();
98 }
99
100 return x;
101 }

114 inline real rand_rejectsamp(
115 stat_function f, const vec_buff& theta,
116 real_function p, real_function Pinv,
117 PRNG& g, unsigned int max_tries = 100) {
118
119 for (unsigned int i = 0; i < max_tries; ++i) {
120
121 // Generate a random number following
122 // the p(x) probability distribution
123 // by the inverse cumulative distribution function
124 const real u_1 = rand_uniform(0, 1, g);
125 const real x_p = Pinv(u_1);
126
127 const real u_2 = rand_uniform(0, 1, g);
128
129 // Accept the sample if f(x_p)/g(x_p) > u_2
130 if(u_2 * p(x_p) < f(x_p, theta))
131 return x_p;
132 }
133
134 TH_MATH_ERROR("rand_reject_sample", max_tries, NO_ALGO_CONVERGENCE);
135 return nan();
136 }

144 inline real rand_gaussian_polar(real mean, real sigma, PRNG& g) {
145
146 static real spare;
147 static bool has_spare = false;
148
149 if(has_spare) {
150 has_spare = false;
151 return mean + spare * sigma;
152 }
153
154 real x, y, s;
155
156 // Generate a random point inside the unit circle
157 do {
158
159 x = rand_uniform(-1, 1, g);
160 y = rand_uniform(-1, 1, g);
161 s = square(x) + square(y);
162
163 } while(s >= 1 || s <= MACH_EPSILON);
164
165 // Project the point
166 s = sqrt(-2 * ln(s) / s);
167
168 // Keep the second generated value for future calls
169 spare = y * s;
170 has_spare = true;
171
172 return mean + sigma * x * s;
173 }

181 inline real rand_gaussian_boxmuller(real mean, real sigma, PRNG& g) {
182
183 static real spare;
184 static bool has_spare = false;
185
186 if(has_spare) {
187 has_spare = false;
188 return mean + spare * sigma;
189 }
190
191 // Generate a random point inside the unit circle
192
193 const real x = rand_uniform(0, 1, g);
194 const real y = rand_uniform(0, 1, g);
195
196 const real x_transf = sqrt(-2 * ln(x));
197
198 const real u = x_transf * cos(TAU * y);
199 const real v = x_transf * sin(TAU * y);
200
201 spare = v;
202 has_spare = true;
203
204 return mean + sigma * u;
205 }

219 inline real rand_gaussian_clt(real mean, real sigma, PRNG& g) {
220
221 // Fixed N = 12
222 constexpr unsigned int N = 12;
223
224 real s = 0;
225 for (unsigned int i = 0; i < N; ++i)
226 s += rand_uniform(-1, 1, g);
227
228 // f(u) = 1/2 (in [-1, 1])
229 // E[u] = 0
230 // sqrt(V[u]) = 1 / sqrt(3N) = 1 / 6
231
232 return mean + (s / static_cast<real>(N)) * sigma * 6;
233 }

501 inline real metropolis(
502 real_function f, pdf_sampler& g,
503 real x0, PRNG& rnd, unsigned int depth = METROPOLIS_DEPTH) {
504
505 real current = x0, next;
506
507 for(unsigned int i = 0; i < depth; i++) {
508
509 // Computes the next step
510 next = current + g();
511
512 // Checks acceptance rate
513 if(rand_uniform(0, 1, rnd) * f(current) <= f(next))
514 current = next;
515 }
516
517 return current;
518 }
