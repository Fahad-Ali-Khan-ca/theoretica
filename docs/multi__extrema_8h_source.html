95 template<unsigned int N>
96 inline vec<N> minimize_lingrad(
97 multidual<N>(*f)(vec<N, multidual<N>>),
98 vec<N> guess = vec<N>(0),
99 real tolerance = MINGRAD_TOLERANCE,
100 unsigned int max_iter = MINGRAD_MAX_ITER) {
101
102 vec<N> x = guess;
103 vec<N> grad;
104 unsigned int iter = 0;
105
106 do {
107
108 grad = gradient(f, x);
109
110 // Minimize f(x + gamma * gradient) in [-1, 0]
111 // using Golden Section extrema search
112 real gamma = approx_min_goldensection(
113 [f, x, grad](real gamma){
114 return f(
115 multidual<N>::pack_function_arg(x)
116 + multidual<N>::pack_function_arg(grad) * gamma).Re();
117 },
118 -1, 0);
119
120 // Fallback value
121 if(-gamma <= MACH_EPSILON)
122 gamma = MINGRAD_GAMMA;
123
124 x += gamma * grad;
125 iter++;
126
127 } while(grad.length() > MINGRAD_TOLERANCE && iter <= max_iter);
128
129 if(iter > max_iter) {
130 TH_MATH_ERROR("minimize_lingrad", iter, NO_ALGO_CONVERGENCE);
131 return vec<N>(nan());
132 }
133
134 return x;
135 }
