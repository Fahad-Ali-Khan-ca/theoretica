100 template<unsigned int N>
101 inline vec<N> minimize_lingrad(
102 multidual<N>(*f)(vec<N, multidual<N>>),
103 vec<N> guess = vec<N>(0),
104 real tolerance = MINGRAD_TOLERANCE,
105 unsigned int max_iter = MINGRAD_MAX_ITER) {
106
107 vec<N> x = guess;
108 vec<N> grad;
109 unsigned int iter = 0;
110
111 do {
112
113 grad = gradient(f, x);
114
115 // Minimize f(x + gamma * gradient) in [-1, 0]
116 // using Golden Section extrema search
117 real gamma = approx_min_goldensection(
118 [f, x, grad](real gamma){
119 return f(
120 multidual<N>::make_argument(x)
121 + multidual<N>::make_argument(grad) * gamma).Re();
122 },
123 -1, 0);
124
125 // Fallback value
126 if(-gamma <= MACH_EPSILON)
127 gamma = MINGRAD_GAMMA;
128
129 x += gamma * grad;
130 iter++;
131
132 } while(grad.length() > MINGRAD_TOLERANCE && iter <= max_iter);
133
134 if(iter > max_iter) {
135 TH_MATH_ERROR("minimize_lingrad", iter, NO_ALGO_CONVERGENCE);
136 return vec<N>(nan());
137 }
138
139 return x;
140 }
