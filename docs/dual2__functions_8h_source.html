37 dual2 pow(dual2 x, int n) {
38
39 real pow_n_2_x = pow(x.Re(), n - 2);
40
41 real f = pow_n_2_x * square(x.Re());
42 real df = pow_n_2_x * x.Re() * n * x.Dual1();
43 real d2f = (pow_n_2_x * n * (n - 1)) * square(x.Dual1()) + df * x.Dual2();
44
45 return dual2(f, df, d2f);
46 }

50 dual2 sqrt(dual2 x) {
51
52 real sqrt_x = sqrt(x.Re());
53
54 if(sqrt_x == 0) {
55 UMATH_ERROR("sqrt(dual2)", sqrt_x, DIV_BY_ZERO);
56 return dual2(nan(), nan());
57 }
58
59 return dual2(sqrt_x,
60 0.5 / sqrt_x * x.Dual1(),
61 -0.25 / x.Re() / sqrt_x * square(x.Dual1()) + 0.5 / sqrt_x * x.Dual2());
62 }

186

188 UMATH_ERROR("asin(dual2)", x.Re(), OUT_OF_DOMAIN);
189 return dual2(nan(), nan());

197 + x.Dual2() / sqrt_1mx2);
