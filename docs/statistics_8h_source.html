139 template<typename Dataset1, typename Dataset2>
140 inline real propagate_product(const Dataset1& sigma, const Dataset2& mean) {
141
142 if(sigma.size() != mean.size()) {
143 TH_MATH_ERROR("propagate_product", sigma.size(), INVALID_ARGUMENT);
144 return nan();
145 }
146
147 // Compute sum of squares of (i_sigma / i_mean)
148 real s = 0;
149 for (unsigned int i = 0; i < sigma.size(); ++i) {
150
151 if(mean[i] == 0) {
152 TH_MATH_ERROR("propagate_product", mean[i], DIV_BY_ZERO);
153 return nan();
154 }
155
156 s += square(sigma[i] / abs(mean[i]));
157 }
158
159 return sqrt(s);
160 }

167 template<typename Dataset>
168 inline real total_sum_squares(const Dataset& X) {
169
170 if(!X.size()) {
171 TH_MATH_ERROR("total_sum_squares", X.size(), INVALID_ARGUMENT);
172 return nan();
173 }
174
175 // Running average
176 real m = X[0];
177
178 // Total sum
179 real s = 0;
180
181 for (size_t i = 1; i < X.size(); ++i) {
182
183 // m_{i - 1}
184 const real m_tmp = m;
185
186 m = m_tmp + (X[i] - m_tmp) / (i + 1);
187 s += (X[i] - m_tmp) * (X[i] - m);
188 }
189
190 return s;
191 }

329 real s = 0;

384 template<typename Dataset>
385 inline real autocorrelation(const Dataset& X, unsigned int n = 1) {
386
387 if(X.size() < n) {
388 TH_MATH_ERROR("autocorrelation", X.size(), INVALID_ARGUMENT);
389 return nan();
390 }
391
392 const real mu = mean(X);
393 real num = 0;
394 real den = square(X[0] - mu);
395
396 for (unsigned int i = n; i < X.size(); ++i) {
397
398 const real delta = X[i] - mu;
399 num += delta * (X[i - n] - mu);
400 den += square(delta);
401 }
402
403 return num / den;
404 }

553 inline real pvalue_chi_squared(real chi_sqr, unsigned int ndf) {
554
555 if(ndf == 0) {
556 TH_MATH_ERROR("pvalue_chi_squared", ndf, INVALID_ARGUMENT);
557 return nan();
558 }
559
560 // For ndf >= 260 use the Gaussian approximation
561 // as the coefficients are not stable
562 if(ndf >= 260) {
563
564 const real new_x = (chi_sqr - ndf) / sqrt(2.0 * ndf);
565
566 // For really low Chi-squared the Gaussian is
567 // below tolerance value for integration
568 if(new_x < 0) {
569
570 if(new_x < -3)
571 return 1 - integral_inf_riemann([=](real x) {
572 return exp(-x * x / 2) / SQRTPI / SQRT2;
573 }, -new_x, 1E-16, 25);
574
575 return 0.5 + integral_romberg_tol([=](real x) {
576 return exp(-x * x / 2) / SQRTPI / SQRT2;
577 }, new_x, 0, 1E-16);
578 } else {
579
580 if(new_x > 3)
581 return integral_inf_riemann([=](real x) {
582 return exp(-x * x / 2) / SQRTPI / SQRT2;
583 }, new_x, 1E-16, 25);
584
585 return 0.5 - integral_romberg_tol([=](real x) {
586 return exp(-x * x / 2) / SQRTPI / SQRT2;
587 }, 0, new_x, 1E-16);
588 }
589 }
590
591 // Compute the coefficient using a stable equivalent formula
592 const real coeff = exp(-special::lngamma(ndf / 2.0) - chi_sqr / 2.0);
593
594 // Use different methods when Gauss-Laguerre is not numerically stable
595 if((ndf > 70 && chi_sqr < (ndf / 2.0))) {
596
597 // Use equivalent formula around potential singularity
598 real res = integral_romberg_tol([=](real x) {
599 return pow(sqrt(x + chi_sqr / 2), ndf - 2) * exp(-x);
600 }, 0, 1, 1E-12);
601
602 res += integral_inf_riemann([=](real x) {
603 return exp((ndf - 2) / 2.0 * ln(x + chi_sqr / 2) - x);
604 }, 1, ndf / 2, 1E-12, 25);
605
606 return coeff * res;
607 }
608
609 // Approximate the integral using Gauss-Laguerre quadrature
610 return coeff * integral_gauss(
611 [=](real x) {
612 return pow(sqrt(x + chi_sqr / 2), ndf - 2);
613 }, tables::laguerre_roots_16, tables::laguerre_weights_16, 16);
614 }

618 template<typename Dataset1, typename Dataset2, typename Dataset3>
619 inline real chi_square_linearization(
620 const Dataset1& X, const Dataset2& Y,
621 const Dataset3& sigma, real intercept, real slope) {
622
623 if(X.size() != Y.size() || X.size() != sigma.size()) {
624 TH_MATH_ERROR(
625 "chi_square_linearization",
626 X.size(), INVALID_ARGUMENT);
627 return nan();
628 }
629
630 real chi_squared = 0;
631 for (unsigned int i = 0; i < X.size(); ++i) {
632
633 if(abs(sigma[i]) <= MACH_EPSILON) {
634 TH_MATH_ERROR("chi_square_linearization", sigma[i], DIV_BY_ZERO);
635 return nan();
636 }
637
638 chi_squared += square((Y[i] - intercept - slope * X[i]) / sigma[i]);
639 }
640
641 return chi_squared;
642 }
