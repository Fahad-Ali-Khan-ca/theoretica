130 inline real propagate_product(const vec_buff& sigma, const vec_buff& mean) {
131
132 if(sigma.size() != mean.size()) {
133 TH_MATH_ERROR("propagate_product", sigma.size(), INVALID_ARGUMENT);
134 return nan();
135 }
136
137 // Compute sum of squares of (i_sigma / i_mean)
138 real s = 0;
139 for (unsigned int i = 0; i < sigma.size(); ++i) {
140
141 if(mean[i] == 0) {
142 TH_MATH_ERROR("propagate_product", mean[i], DIV_BY_ZERO);

144 }

146 s += square(sigma[i] / abs(mean[i]));

148
149 return sqrt(s);

286 inline real covariance(const vec_buff& X, const vec_buff& Y) {
287
288 if(X.size() != Y.size()) {
289 TH_MATH_ERROR("covariance", X.size(), INVALID_ARGUMENT);
290 return nan();
291 }
292
293 real s = 0;
294 real X_mean = mean(X);
295 real Y_mean = mean(Y);
296
297 for (unsigned int i = 0; i < X.size(); ++i)
298 s += (X[i] - X_mean) * (Y[i] - Y_mean);
299
300 return s / (real) X.size();
301 }

341 inline real autocorrelation(const vec_buff& X, unsigned int n = 1) {
342
343 if(X.size() < n) {
344 TH_MATH_ERROR("autocorrelation", X.size(), INVALID_ARGUMENT);
345 return nan();
346 }
347
348 const real mu = mean(X);
349 real num = 0;
350 real den = square(X[0] - mu);
351
352 for (unsigned int i = n; i < X.size(); ++i) {
353
354 const real delta = X[i] - mu;
355 num += delta * (X[i - n] - mu);
356 den += square(delta);
357 }
358
359 return num / den;
360 }

418 inline real chi_square(
419 const vec_buff& O,
420 const vec_buff& E,
421 const vec_buff& sigma) {
422
423 if(O.size() != E.size() || E.size() != sigma.size()) {
424 TH_MATH_ERROR("chi_square", E.size(), INVALID_ARGUMENT);
425 return nan();
426 }
427
428 real c_sqr = 0;
429
430 for (unsigned int i = 0; i < O.size(); ++i) {

433 TH_MATH_ERROR("chi_square", sigma[i], DIV_BY_ZERO);
434 return nan();
435 }
436
437 c_sqr += square((O[i] - E[i]) / sigma[i]);
438 }
439
440 return c_sqr;

454 inline real pvalue_chi_squared(real chi_sqr, unsigned int ndf) {
455
456 // Compute the coefficient using a stable equivalent formula
457 const real coeff = exp(-special::lngamma(ndf / 2.0) - chi_sqr / 2.0);
