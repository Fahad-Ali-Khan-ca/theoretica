127 inline real propagate_product(const vec_buff& sigma, const vec_buff& mean) {
128
129 if(sigma.size() != mean.size()) {
130 TH_MATH_ERROR("propagate_product", sigma.size(), INVALID_ARGUMENT);
131 return nan();
132 }
133
134 // Compute sum of squares of (i_sigma / i_mean)
135 real s = 0;
136 for (unsigned int i = 0; i < sigma.size(); ++i) {
137
138 if(mean[i] == 0) {
139 TH_MATH_ERROR("propagate_product", mean[i], DIV_BY_ZERO);
140 return nan();
141 }
142
143 s += square(sigma[i] / abs(mean[i]));
144 }
145
146 return sqrt(s);
147 }

283 inline real covariance(const vec_buff& X, const vec_buff& Y) {
284
285 if(X.size() != Y.size()) {
286 TH_MATH_ERROR("covariance", X.size(), INVALID_ARGUMENT);
287 return nan();
288 }
289
290 real s = 0;
291 real X_mean = mean(X);
292 real Y_mean = mean(Y);
293
294 for (unsigned int i = 0; i < X.size(); ++i)
295 s += (X[i] - X_mean) * (Y[i] - Y_mean);
296
297 return s / (real) X.size();
298 }

335 inline real autocorrelation(const vec_buff& X) {
336
337 if(X.size() < 2) {
338 TH_MATH_ERROR("autocorrelation", X.size(), INVALID_ARGUMENT);
339 return nan();
340 }
341
342 real mu = mean(X);
343 real num = 0;
344 real den = square(X[0] - mu);
345
346 for (unsigned int i = 1; i < X.size(); ++i) {
347
348 const real delta = X[i] - mu;
349 num += delta * (X[i - 1] - mu);
350 den += square(delta);
351 }
352
353 return num / den;
354 }

438 inline real chi_square_sigma(const vec_buff& X) {
439
440 if(!X.size()) {
441 TH_MATH_ERROR("chi_square_sigma", X.size(), INVALID_ARGUMENT);
442 return nan();
443 }
444
445 const unsigned int N = X.size();
446
447 unsigned int Ok_1 = 0;
448 unsigned int Ok_2 = 0;
449 unsigned int Ok_3 = 0;
450 unsigned int Ok_4 = 0;
451
452 real M = mean(X);
453 real sigma = sample_standard_deviation(X);
454
455 for (auto x : X) {
456 if(x < M - sigma)
457 Ok_1++;
458 else if((x > M - sigma) && (x < M))
459 Ok_2++;
460 else if((x > M) && (x < M + sigma))
461 Ok_3++;
462 else if(x > M + sigma)
463 Ok_4++;
464 }
465
466 // Sum of (Ok - Ek)^2 / Ek
467 // where Ek = N * Pk
468 real chi_2 = ((Ok_1 - (N * 0.16)) * (Ok_1 - (N * 0.16))) / (N * 0.16);
469 chi_2 += ((Ok_2 - (N * 0.34)) * (Ok_2 - (N * 0.34))) / (N * 0.34);
470 chi_2 += ((Ok_3 - (N * 0.34)) * (Ok_3 - (N * 0.34))) / (N * 0.34);
471 chi_2 += ((Ok_4 - (N * 0.16)) * (Ok_4 - (N * 0.16))) / (N * 0.16);
472
473 return chi_2;
474 }

541 const vec_buff& X, const vec_buff& Y,
542 real intercept, real slope) {
543
544 if(X.size() != Y.size()) {

548
549 real err = 0;
550 for (unsigned int i = 0; i < X.size(); ++i) {
551 err += square(Y[i] - intercept - slope * X[i]);
