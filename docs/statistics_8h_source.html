251 inline real covariance(const vec_buff& X, const vec_buff& Y) {
252
253 if(X.size() != Y.size()) {
254 UMATH_ERROR("covariance", X.size(), INVALID_ARGUMENT);
255 return nan();
256 }
257
258 real sum = 0;
259 real X_mean = mean(X);
260 real Y_mean = mean(Y);
261
262 for (int i = 0; i < X.size(); ++i)
263 sum += (X[i] - X_mean) * (Y[i] - Y_mean);
264
265 return sum / (real) X.size();
266 }

304 inline real chi_square_sigma(const vec_buff& X) {
305
306 if(!X.size()) {
307 UMATH_ERROR("chi_square_sigma", X.size(), INVALID_ARGUMENT);
308 return nan();
309 }
310
311 const unsigned int N = X.size();
312
313 unsigned int Ok_1 = 0;
314 unsigned int Ok_2 = 0;
315 unsigned int Ok_3 = 0;
316 unsigned int Ok_4 = 0;
317
318 real M = mean(X);
319 real sigma = sample_standard_deviation(X);
320
321 for (auto x : X) {
322 if(x < M - sigma)
323 Ok_1++;
324 else if((x > M - sigma) && (x < M))
325 Ok_2++;
326 else if((x > M) && (x < M + sigma))
327 Ok_3++;
328 else if(x > M + sigma)
329 Ok_4++;
330 }
331
332 // Sum of (Ok - Ek)^2 / Ek
333 // where Ek = N * Pk
334 real chi_2 = ((Ok_1 - (N * 0.16)) * (Ok_1 - (N * 0.16))) / (N * 0.16);
335 chi_2 += ((Ok_2 - (N * 0.34)) * (Ok_2 - (N * 0.34))) / (N * 0.34);
336 chi_2 += ((Ok_3 - (N * 0.34)) * (Ok_3 - (N * 0.34))) / (N * 0.34);
337 chi_2 += ((Ok_4 - (N * 0.16)) * (Ok_4 - (N * 0.16))) / (N * 0.16);
338
339 return chi_2;
340 }

344 inline real least_squares_linear_intercept(
345 const vec_buff& X, const vec_buff& Y) {
346
347 if(X.size() != Y.size()) {
348 UMATH_ERROR("least_squares_linear_intercept", X.size(), INVALID_ARGUMENT);
349 return nan();
350 }
351
352 real Delta = X.size() * sum_squares(X) - square(sum(X));
353 real A = (sum_squares(X) * sum(Y) - sum(X) * product_sum(X, Y)) / Delta;
354
355 return A;

406 inline real least_squares_linear_error(
407 const vec_buff& X, const vec_buff& Y,
408 real intercept, real slope) {
409
410 if(X.size() != Y.size()) {
411 UMATH_ERROR("least_squares_linear_error", X.size(), INVALID_ARGUMENT);
412 return nan();
413 }
414
415 real err = 0;
416 for (int i = 0; i < X.size(); ++i) {
417 err += square(Y[i] - intercept - slope * X[i]);
418 }

420 // Correction by degrees of freedom (N - 2)
421 return sqrt(err / (real) (X.size() - 2));
