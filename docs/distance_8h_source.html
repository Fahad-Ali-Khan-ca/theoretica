136 template<typename Vector, typename T = real>
137 inline auto hermitian_distance(const Vector& v1, const Vector& v2) {
138
139 if(v1.size() != v2.size()) {
140 TH_MATH_ERROR("hermitian_distance", v1.size(), INVALID_ARGUMENT);
141 return complex<T>(nan());
142 }
143
144 complex<T> sum = 0;
145
146 for (size_t i = 0; i < v1.size(); ++i) {
147 const complex<T> diff = v1.get(i) - conjugate(v2.get(i));
148 sum += diff * diff.conjugate();
149 }
150
151 return sqrt(sum);
152 }

191 template<typename Vector>
192 inline real discrete_distance(
193 const Vector& v1, const Vector& v2, real tolerance = MACH_EPSILON) {
194
195 if(v1.size() != v2.size()) {
196 TH_MATH_ERROR("discrete_distance", v1.size(), INVALID_ARGUMENT);
197 return nan();
198 }
199
200 bool diff = false;
201
202 for (size_t i = 0; i < v1.size(); ++i) {
203
204 // The vectors differ if a pair of elements differs
205 // more than the given tolerance
206 if(abs(v1.get(i) - v2.get(i)) > tolerance) {
207 diff = true;
208 break;
209 }
210 }
211
212 return diff ? 1 : 0;
213 }

235 template<typename Vector>
236 inline real cosine_distance(const Vector& v1, const Vector& v2) {
237
238 if(v1.size() != v2.size()) {
239 TH_MATH_ERROR("cosine_distance", v1.size(), INVALID_ARGUMENT);
240 return nan();
241 }
242
243 real product = 0;
244 real sum_sqr_x = 0;
245 real sum_sqr_y = 0;
246
247 for (size_t i = 0; i < v1.size(); ++i) {
248 product += v1.get(i) * v2.get(i);
249 sum_sqr_x += square(v1.get(i));
250 sum_sqr_y += square(v2.get(i));
251 }
252
253 return product / sqrt(sum_sqr_x * sum_sqr_y);
254 }
