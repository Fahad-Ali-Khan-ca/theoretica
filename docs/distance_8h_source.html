135 template<typename Vector>
136 inline complex hermitian_distance(Vector v1, Vector v2) {
137
138 if(v1.size() != v2.size()) {
139 TH_MATH_ERROR("hermitian_distance", v1.size(), INVALID_ARGUMENT);
140 return nan();
141 }
142
143 complex sum = 0;
144
145 for (size_t i = 0; i < v1.size(); ++i) {
146 const complex diff = v1[i] - conjugate(v2[i]);
147 sum += diff * diff.conjugate();
148 }
149
150 return sqrt(sum);
151 }

188 template<typename Vector>
189 inline real discrete_distance(Vector v1, Vector v2, real tolerance = MACH_EPSILON) {
190
191 if(v1.size() != v2.size()) {
192 TH_MATH_ERROR("discrete_distance", v1.size(), INVALID_ARGUMENT);
193 return nan();
194 }
195
196 bool diff = false;
197
198 for (size_t i = 0; i < v1.size(); ++i) {
199
200 // The vectors differ if a pair of elements differs
201 // more than the given tolerance
202 if(abs(v1[i] - v2[i]) > tolerance) {
203 diff = true;
204 break;
205 }
206 }
207
208 return diff ? 1 : 0;
209 }

231 template<typename Vector>
232 inline real cosine_distance(Vector v1, Vector v2) {
233
234 if(v1.size() != v2.size()) {
235 TH_MATH_ERROR("cosine_distance", v1.size(), INVALID_ARGUMENT);
236 return nan();
237 }
238
239 real sum_sqr_x = 0;
240 real sum_sqr_y = 0;
241
242 for (size_t i = 0; i < v1.size(); ++i) {
243 sum_sqr_x += square(v1[i]);
244 sum_sqr_y += square(v2[i]);
245 }
246
247 return (v1 * v2) / sqrt(sum_sqr_x * sum_sqr_y);
248 }
