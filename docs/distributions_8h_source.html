52 inline real gaussian(real x, real X, real sigma) {
53
54 return (1.0 / (sigma * SQRT2 * SQRTPI))
55 * exp(-square(x - X) / (2 * square(sigma)));

133 inline real multinomial(
134 std::vector<unsigned int> x,
135 unsigned int n,
136 unsigned int k,
137 std::vector<real> p) {
138
139 if(x.size() != p.size() || x.size() != k) {
140 TH_MATH_ERROR("distribution::multinomial", x.size(), INVALID_ARGUMENT);
141 return nan();
142 }
143
144 real res = fact(n);
145
146 for (unsigned int i = 0; i < k; ++i)
147 res *= pow(p[i], x[i]) / fact(x[i]);
148
149 return res;
150 }

174 inline real student(real x, unsigned int nu) {
175
176 real a = 1 + (x * x / nu);
177
178 return (1.0 / SQRTPI) * (1.0 / sqrt(nu))
179 * (special::gamma((nu + 1) / 2.0) / special::gamma(nu / 2.0))
180 * pow(sqrt(a), -nu - 1);
181 }
