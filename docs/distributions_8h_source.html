58 inline real gaussian(real x, real X, real sigma) {
59
60 return (1.0 / (sigma * SQRT2 * SQRTPI))

138 inline real multinomial(
139 std::vector<unsigned int> x,
140 unsigned int n,
141 unsigned int k,
142 std::vector<real> p) {
143
144 if(x.size() != p.size() || x.size() != k) {
145 TH_MATH_ERROR("distribution::multinomial", x.size(), INVALID_ARGUMENT);
146 return nan();
147 }
148
149 real res = fact(n);
150
151 for (unsigned int i = 0; i < k; ++i)
152 res *= pow(p[i], x[i]) / fact(x[i]);
153
154 return res;

159 inline real chi_squared(real x, unsigned int k) {
160
161 if(x < 0)
162 return 0;
163
164 const real coeff = special::half_gamma(k);
165
166 if(k % 2 == 0)
167 return pow(x, int(k / 2) - 1) * exp(-x / 2.0)
168 / (pow(SQRT2, k) * coeff);
169 else
170 return pow(sqrt(x), k - 2) * exp(-x / 2.0)
171 / (pow(SQRT2, k) * coeff);
172 }
