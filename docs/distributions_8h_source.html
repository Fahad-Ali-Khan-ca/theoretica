60

139 inline real multinomial(
140 std::vector<unsigned int> x,
141 unsigned int n,
142 unsigned int k,
143 std::vector<real> p) {
144
145 if(x.size() != p.size() || x.size() != k) {
146 TH_MATH_ERROR("distribution::multinomial", x.size(), INVALID_ARGUMENT);
147 return nan();
148 }
149
150 real res = fact(n);
151
152 for (unsigned int i = 0; i < k; ++i)
153 res *= pow(p[i], x[i]) / fact(x[i]);
154
155 return res;

160 inline real chi_squared(real x, unsigned int k) {
161
162 if(x < 0)
163 return 0;
164
165 const real coeff = special::half_gamma(k);
166
167 if(k % 2 == 0)
168 return pow(x, int(k / 2) - 1) * exp(-x / 2.0)
169 / (pow(SQRT2, k) * coeff);
170 else
171 return pow(sqrt(x), k - 2) * exp(-x / 2.0)
172 / (pow(SQRT2, k) * coeff);
173 }

232

234

236 * pow(sqrt(a), -nu - 1) / (sqrt(nu) * SQRTPI);
