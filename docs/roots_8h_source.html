24 template<typename RealFunction>
25 inline std::vector<vec2> find_root_intervals(RealFunction f, real a, real b, unsigned int steps = 10) {
26
27 std::vector<vec2> res;
28 const real dx = (b - a) / (real) steps;
29
30 for (unsigned int i = 0; i < steps - 1; ++i) {
31
32 const real x1 = a + i * dx;
33 const real x2 = a + (i + 1) * dx;
34
35 if(f(x1) * f(x2) <= 0)
36 res.push_back({x1, x2});
37 }
38
39 return res;
40 }

45 template<typename RealFunction>
46 inline real approx_root_bisection(RealFunction f, real a, real b) {
47
48 if(f(a) * f(b) >= 0) {
49 TH_MATH_ERROR("approx_root_bisection", f(a) * f(b), INVALID_ARGUMENT);
50 return nan();
51 }
52
53 real x_avg = a;
54
55 real x_min = a;
56 real x_max = b;
57
58 unsigned int iter = 0;
59
60 while((x_max - x_min) > BISECTION_APPROX_TOL && iter <= MAX_BISECTION_ITER) {
61
62 x_avg = (x_max + x_min) / 2.0;
63
64 if(f(x_avg) * f(x_min) > 0)
65 x_min = x_avg;
66 else
67 x_max = x_avg;
68
69 ++iter;
70 }
71
72 if(iter > MAX_BISECTION_ITER) {
73 TH_MATH_ERROR("approx_root_bisection", x_avg, NO_ALGO_CONVERGENCE);
74 return nan();
75 }
76
77 return x_avg;
78 }

342 template<typename RealFunction>
343 inline std::vector<real> roots(RealFunction f, real a, real b, real steps = 10) {
344
345 // Find candidate intervals
346 std::vector<vec2> intervals = find_root_intervals(f, a, b, steps);
347
348 std::vector<real> res;
349 res.reserve(intervals.size());
350
351 // Iterate through all candidate intervals and refine the results
352 for (unsigned int i = 0; i < intervals.size(); ++i) {
353
354 // Check whether the extremes of the candidate intervals
355 // happen to coincide with the roots
356 if(abs(f(intervals[i][0])) <= MACH_EPSILON) {
357 res.push_back(intervals[i][0]);
358 continue;
359 }
360
361 if(abs(f(intervals[i][1])) <= MACH_EPSILON) {
362 res.push_back(intervals[i][1]);
363 continue;
364 }
365
366 // Approximate the roots using bisection inside each interval
367 res.push_back(
368 approx_root_bisection(f, intervals[i][0], intervals[i][1]));
369 }
370
371 return res;
372 }
