19 inline real approx_root_bisection(real_function f, real a, real b) {
20
21 if(f(a) * f(b) >= 0) {
22 UMATH_ERROR("approx_root_bisection", f(a) * f(b), INVALID_ARGUMENT);
23 return nan();
24 }
25
26 real x_avg = a;
27
28 real x_min = a;
29 real x_max = b;
30
31 int iter = 0;
32
33 while(x_max - x_min > BISECTION_APPROX_TOL && iter <= MAX_BISECTION_ITER) {
34
35 x_avg = (x_max + x_min) / 2.0;
36
37 if(f(x_avg) * f(x_min) > 0)
38 x_min = x_avg;
39 else
40 x_max = x_avg;
41
42 ++iter;
43 }
44
45 if(iter > MAX_BISECTION_ITER) {
46 UMATH_ERROR("approx_root_bisection", x_avg, NO_ALGO_CONVERGENCE);
47 return nan();
48 }
49
50 return x_avg;
51 }

55 inline real approx_root_newton(real_function f, real_function Df, real guess = 0) {

57
