49 template<
50 unsigned int N = 0,
51 typename MultiDualFunction = d_real<N>(*)(d_vec<N>)>
52 inline real error_propagation(
53 MultiDualFunction f,
54 const vec<real, N>& x_best, const vec<real, N>& delta_x) {
55
56 real err_sqr = 0;
57 const multidual<N> df = f(multidual<N>::make_argument(x_best));
58
59 for (unsigned int i = 0; i < x_best.size(); ++i)
60 err_sqr += square(df.Dual().get(i) * delta_x.get(i));
61
62 return sqrt(err_sqr);
63 }

79 template <
80 unsigned int N = 0, unsigned int M = 0,
81 typename MultiDualFunction = d_real<N>(*)(d_vec<N>)>
82 inline real error_propagation(
83 MultiDualFunction f,
84 const vec<real, N>& x_best, const mat<real, M, M>& cm) {
85
86
87 if(cm.rows() != x_best.size()) {
88 TH_MATH_ERROR("error_propagation", cm.rows(), INVALID_ARGUMENT);
89 return nan();
90 }
91
92 if(cm.cols() != x_best.size()) {
93 TH_MATH_ERROR("error_propagation", cm.cols(), INVALID_ARGUMENT);
94 return nan();
95 }
96
97 real err_sqr = 0;
98 const multidual<N> df = f(multidual<N>::make_argument(x_best));
99
100 for (unsigned int i = 0; i < cm.rows(); ++i)
101 for (unsigned int j = 0; j < cm.cols(); ++j)
102 err_sqr += df.Dual().get(i) * df.Dual().get(j)
103 * cm(i, j);
104
105 return sqrt(err_sqr);
106 }
