51 template<unsigned int N>
52 inline real propagate_err(
53 multidual<N>(*f)(vec<N, multidual<N>>),
54 const vec<N>& x_best, const vec<N>& delta_x) {
55
56 real err_sqr = 0;
57 const multidual<N> df = f(multidual<N>::make_argument(x_best));
58
59 for (unsigned int i = 0; i < N; ++i)
60 err_sqr += square(df.Dual().get(i) * delta_x.get(i));
61
62 return sqrt(err_sqr);

79 template<unsigned int N>
80 inline real propagate_err(
81 multidual<N>(*f)(vec<N, multidual<N>>),

83
84 real err_sqr = 0;
85 const multidual<N> df = f(multidual<N>::make_argument(x_best));
86
87 for (unsigned int i = 0; i < N; ++i)
88 for (unsigned int j = 0; j < N; ++j)
89 err_sqr += df.Dual().get(i) * df.Dual().get(j)
