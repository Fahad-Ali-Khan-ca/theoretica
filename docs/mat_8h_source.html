530 inline real det_gj() const {
531
532 if(!is_square()) {
533 UMATH_ERROR("mat::det_gj", N, IMPOSSIBLE_OPERATION);
534 return nan();
535 }
536
537 mat<N, K> A = mat<N, K>(*this);
538
539 // Iterate on all columns
540 for (unsigned int i = 0; i < N; ++i) {
541
542 // Make sure the element on the diagonal
543 // is non-zero by adding the first non-zero row
544 if(A.iat(i, i) == 0) {
545
546 bool flag = false;
547
548 // Iterate on all rows
549 for (unsigned int j = i + 1; j < N; ++j) {
550
551 // Add the j-th row to the i-th row
552 // if Aji is non-zero.
553 // The determinant does not change
554 // when adding a row to another one
555 if(A.iat(j, i) != 0) {
556
557 for (unsigned int k = 0; k < N; ++k) {
558 A.iat(i, k) += A.iat(j, k);
559 }
560
561 flag = true;
562 break;
563 }
564 }
565
566 if(!flag) {
567 return 0;
568 }
569 }
570
571 real inv_pivot = 1.0 / A.iat(i, i);
572
573 // Use the current row to make all other
574 // elements of the column equal to zero
575 for (unsigned int j = i + 1; j < N; ++j) {
576
577 // Multiplication coefficient for
578 // the elision of Ajk
579 real coeff = A.iat(j, i) * inv_pivot;
580
581 // The coefficient does not change
582 // when adding a linear combination
583 // of a row to another
584 for (unsigned int k = 0; k < N; ++k) {
585 A.iat(j, k) -= coeff * A.iat(i, k);
586 }
587 }
588 }
589
590 // The determinant of a (lower) triangular matrix
591 // is the product of the elements on its diagonal
592 return A.diagonal_product();
593 }

618 inline mat<N, K> inverse_2x2() const {
619
620 mat<N, K> B;
621
622 // Exchange elements on the diagonal
623 B.iat(0, 0) = iget(1, 1);
624 B.iat(1, 1) = iget(0, 0);
625
626 // Change sign of the other elements
627 B.iat(1, 0) = -iget(1, 0);
628 B.iat(0, 1) = -iget(0, 1);
629
630 return B / B.det();
631 }

635 inline mat<N, K> inverse() const {
636
637 if(!is_square()) {
638 UMATH_ERROR("mat::inverse", N, IMPOSSIBLE_OPERATION);
639 return mat<N, K>(nan());
640 }
641
642 if(N == 2)
643 return inverse_2x2();
644
645 // Initialize the needed matrices
646 // (A|B) is the augmented matrix
647 mat<N, K> A = mat<N, K>(*this);
648 mat<N, K> B = mat<N, K>::identity();
649
650 // Gauss-Jordan elimination
651
652 // Iterate on all columns
653 for (unsigned int i = 0; i < N; ++i) {
654
655 // Make sure the element on the diagonal
656 // is non-zero by adding the first non-zero row
657 if(A.iat(i, i) == 0) {
658
659 bool flag = false;
660
661 // Iterate on all rows
662 for (unsigned int j = i + 1; j < N; ++j) {
663
664 // Add the j-th row to the i-th row
665 // if Aji is non-zero
666 if(A.iat(j, i) != 0) {
667
668 for (unsigned int k = 0; k < N; ++k) {
669 A.iat(i, k) += A.iat(j, k);
670 B.iat(i, k) += B.iat(j, k);
671 }
672
673 flag = true;
674 break;
675 }
676 }
677
678 if(!flag) {
679 UMATH_ERROR("mat::inverse", flag, IMPOSSIBLE_OPERATION);
680 return mat<N, K>(nan());
681 }
682 }
683
684 real inv_pivot = 1.0 / A.iat(i, i);
685
686 // Use the current row to make all other
687 // elements of the column equal to zero
688 for (unsigned int j = 0; j < N; ++j) {
689
690 // Skip the current row
691 if(j == i)
692 continue;
693
694 // Multiplication coefficient for
695 // the elision of Ajk
696 real coeff = A.iat(j, i) * inv_pivot;
697
698 for (unsigned int k = 0; k < N; ++k) {
699 A.iat(j, k) -= coeff * A.iat(i, k);
700 B.iat(j, k) -= coeff * B.iat(i, k);
701 }
702 }
703
704 // Divide the current row by the pivot
705 for (unsigned int j = 0; j < N; ++j) {
706 A.iat(i, j) *= inv_pivot;
707 B.iat(i, j) *= inv_pivot;
708 }
709
710 }
711
712 return B;
713 }

717 inline mat<N, K>& invert() {
718
719 if(!is_square()) {
720 UMATH_ERROR("mat::invert", N, IMPOSSIBLE_OPERATION);
721 return *this;
722 }
723
724 if(N == 2) {
725 *this = inverse_2x2();
726 }
727
728 // Initialize the needed matrices
729 // (A|B) is the augmented matrix
730 mat<N, K> A = mat<N, K>(*this);
731 mat<N, K> B = mat<N, K>::identity();
732
733 // Gauss-Jordan elimination
734
735 // Iterate on all columns
736 for (unsigned int i = 0; i < N; ++i) {
737
738 // Make sure the element on the diagonal
739 // is non-zero by adding the first non-zero row
740 if(A.iat(i, i) == 0) {
741
742 bool flag = false;
743
744 // Iterate on all rows
745 for (unsigned int j = i + 1; j < N; ++j) {
746
747 // Add the j-th row to the i-th row
748 // if Aji is non-zero
749 if(A.iat(j, i) != 0) {
750
751 for (unsigned int k = 0; k < N; ++k) {
752 A.iat(i, k) += A.iat(j, k);
753 B.iat(i, k) += B.iat(j, k);
754 }
755
756 flag = true;
757 break;
758 }
759 }
760
761 if(!flag) {
762 UMATH_ERROR("mat::invert", flag, IMPOSSIBLE_OPERATION);
763 return *this;
764 }
765 }
766
767 real inv_pivot = 1.0 / A.iat(i, i);
768
769 // Use the current row to make all other
770 // elements of the column equal to zero
771 for (unsigned int j = 0; j < N; ++j) {
772
773 // Skip the current row
774 if(j == i)
775 continue;
776
777 // Multiplication coefficient for
778 // the elision of Ajk
779 real coeff = A.iat(j, i) * inv_pivot;
780
781 for (unsigned int k = 0; k < N; ++k) {
782 A.iat(j, k) -= coeff * A.iat(i, k);
783 B.iat(j, k) -= coeff * B.iat(i, k);
784 }
785 }
786
787 // Divide the current row by the pivot
788 for (unsigned int j = 0; j < N; ++j) {
789 A.iat(i, j) *= inv_pivot;
790 B.iat(i, j) *= inv_pivot;
791 }
792
793 }
794
795 // Modify the matrix only when the inversion
796 // has succeeded
797 *this = B;
798 }

1109 inline static mat<4, 4> lookAt(vec3 camera, vec3 target, vec3 up) {
1110
1111 // Construct an orthonormal basis
1112 vec3 z_axis = (target - camera).normalized();
1113 vec3 x_axis = cross(z_axis, up).normalized();
1114 vec3 y_axis = cross(x_axis, z_axis); // Already normalized
1115
1116 // Negate z_axis to have a right-handed system
1117 z_axis = -z_axis;
1118
1119 // Construct the rotation and translation matrix
1120 mat<4, 4> res;
1121
1122 res.iat(0, 0) = x_axis[0];
1123 res.iat(0, 1) = x_axis[1];
1124 res.iat(0, 2) = x_axis[2];
1125 res.iat(0, 3) = -dot(x_axis, camera);
1126
1127 res.iat(1, 0) = y_axis[0];
1128 res.iat(1, 1) = y_axis[1];
1129 res.iat(1, 2) = y_axis[2];
1130 res.iat(1, 3) = -dot(y_axis, camera);
1131
1132 res.iat(2, 0) = z_axis[0];
1133 res.iat(2, 1) = z_axis[1];
1134 res.iat(2, 2) = z_axis[2];
1135 res.iat(2, 3) = -dot(z_axis, camera);
1136
1137 res.iat(3, 0) = 0;
1138 res.iat(3, 1) = 0;
1139 res.iat(3, 2) = 0;
1140 res.iat(3, 3) = 1;
1141
1142 return res;

1147 inline static mat<N, K> symplectic() {
1148
1149 if(N != K || (N % 2 != 0)) {
1150 UMATH_ERROR("mat::symplectic", N, IMPOSSIBLE_OPERATION);
1151 return mat<N, K>(nan());
1152 }

1154 mat<N, K> res = mat<N, K>();
1155
1156 for (unsigned int i = 0; i < N / 2; ++i) {
1157 for (unsigned int j = N / 2; j < N; ++j) {
1158 if(i == (j - N / 2))
1159 res.iat(i, j) = 1;
1160 }
1161 }
1162
1163 for (unsigned int i = N / 2; i < N; ++i) {
1164 for (unsigned int j = 0; j < N / 2; ++j) {
1165 if((i - N / 2) == (j))
1166 res.iat(i, j) = -1;
1167 }
1168 }
1169
1170 return res;
