640 inline std::string to_string(
641 std::string separator = ", ", bool parenthesis = true) const {
642
643 std::stringstream res;
644
645 for (unsigned int i = 0; i < rows(); ++i) {
646
647 if(parenthesis)
648 res << "(";
649
650 for (unsigned int j = 0; j < cols(); ++j) {
651
652 if(j)
653 res << separator;
654
655 if(abs(get(i, j)) < MACH_EPSILON)
656 res << "0";
657 else
658 res << get(i, j);
659 }
660
661 if(parenthesis)
662 res << ")" << std::endl;
663 }
664
665 return res.str();
666 }

1190

1192 if(rows() == n && cols() == k)
1193 return *this;
1194
1195 // Distinguish between row-first and column-first
1196 // allocation
1197#ifdef THEORETICA_ROW_FIRST
1198 size_t size1 = n, size2 = k;
1199#else
1200 size_t size1 = k, size2 = n;
1201#endif
1202
1203 // The matrix must be allocated anew
1204 if(!data.size()) {
1205
1206 data.resize(size1);
1207 for (unsigned int i = 0; i < size1; ++i)
1208 data[i].resize(size2);
1209
1210 row_sz = n;
1211 col_sz = k;
1212 algebra::mat_zeroes(*this);
1213
1214 // The matrix must be reallocated
1215 } else {
1216
1217 std::vector<std::vector<Type>> new_data;
1218 new_data.resize(size1);
1219 for (unsigned int i = 0; i < size1; ++i)
1220 new_data[i].resize(size2);
1221
1222 // Copy data to new memory location
1223
1224 // Bounds on the region to copy
1225 const size_t row_bound = min(rows(), n);
1226 const size_t col_bound = min(cols(), n);
1227
1228 for (unsigned int i = 0; i < row_bound; ++i) {
1229 for (unsigned int j = 0; j < col_bound; ++j) {
1230#ifdef THEORETICA_ROW_FIRST
1231 new_data[i][j] = get(i, j);
1232#else
1233 new_data[j][i] = get(i, j);
1234#endif
1235 }
1236 }
1237
1238 // If the new matrix size is bigger,
1239 // zero out all remaining entries
1240 for (unsigned int i = 0; i < n; ++i) {
1241 for (unsigned int j = col_bound; j < k; ++j) {
1242
1243
1244#ifdef THEORETICA_ROW_FIRST
1245 new_data[i][j] = (Type) 0;
1246#else
1247 new_data[j][i] = (Type) 0;
1248#endif
1249 }
1250 }
1251
1252
1253 // If the new matrix size is bigger,
1254 // zero out all remaining entries
1255 for (unsigned int i = row_bound; i < n; ++i) {
1256 for (unsigned int j = 0; j < k; ++j) {
1257

1259#ifdef THEORETICA_ROW_FIRST
1260 new_data[i][j] = (Type) 0;
1261#else
1262 new_data[j][i] = (Type) 0;
