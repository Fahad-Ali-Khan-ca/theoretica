37
38 real t = invlerp(P1.get(0), P2.get(0), value);

40 // Check that all computed t_i are the same

76 template<unsigned int N>
77 inline vec<N> slerp(vec<N> P1, vec<N> P2, real t) {
78
79 // Compute (only once) the length
80 // of the input vectors
81 real P1_l = P1.length();
82 real P2_l = P2.length();
83
84 // Check whether one of the vectors is null,
85 // which would make the computation impossible
86 if(P1_l == 0 || P2_l == 0) {
87 TH_MATH_ERROR("slerp", P1_l ? P2_l : P1_l, IMPOSSIBLE_OPERATION);
88 return vec<N>(nan());
89 }
90
91 // Angle between P1 and P2 (from the dot product)
92 real omega = acos((P1 * P2) / (P1_l * P2_l));
93 real s = sin(omega);
94
95 // Check that the sine of the angle is not zero
96 if(s == 0) {
97 TH_MATH_ERROR("slerp", s, DIV_BY_ZERO);
98 return vec<N>(nan());
99 }
100
101 return (P1 * sin((1 - t) * omega) + P2 * sin(t * omega)) / s;
102 }

109 inline real smoothstep(real x1, real x2, real interp) {
110
111 if(x1 == x2) {
112 TH_MATH_ERROR("smoothstep", x1, DIV_BY_ZERO);
113 return nan();
114 }
115
116 // Clamp x between 0 and 1
117 const real x = clamp((interp - x1) / (x2 - x1), 0.0, 1.0);
118
119 // 3x^2 - 2x^3
120 return x * x * (3 - 2 * x);
121 }

125 inline real smootherstep(real x1, real x2, real interp) {
126
127 if(x1 == x2) {
128 TH_MATH_ERROR("smootherstep", x1, DIV_BY_ZERO);
129 return nan();
130 }
131
132 // Clamp x between 0 and 1
133 const real x = clamp((interp - x1) / (x2 - x1), 0.0, 1.0);
134
135 // 6x^5 - 15x^4 + 10x^3
136 return x * x * x * (x * (x * 6 - 15) + 10);
137 }

175 template<unsigned int N>
176 inline vec<N> bezier(std::vector<vec<N>> points, real t) {
177
178 if(points.size() < 2) {
179 TH_MATH_ERROR("bezier", points.size(), INVALID_ARGUMENT);
180 return vec<N>(nan());
181 }
182
183 if(t < 0 || t > 1) {
184 TH_MATH_ERROR("bezier", t, INVALID_ARGUMENT);
185 return vec<N>(nan());
186 }
187
188 for (int index = points.size(); index > 1; --index) {
189
190 for (int i = 0; i < index - 1; ++i)
191 points[i] = lerp(points[i], points[i + 1], t);
192 }
193
194 return points[0];
195 }
