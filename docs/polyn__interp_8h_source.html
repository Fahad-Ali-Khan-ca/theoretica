23 template<typename T>
24 inline polynomial<T> lagrange_polynomial(const std::vector<vec<2, T>>& points) {
25
26 // Check that all x_i are different to prevent
27 // division by zero
28 for (int i = 0; i < points.size() - 1; ++i) {
29 if(points[i].get(0) == points[i + 1].get(0)) {
30 UMATH_ERROR("lagrange_polynomial", points[i].get(0), INVALID_ARGUMENT);
31 return polynomial<T>({T(nan())});
32 }
33 }
34
35 // Lagrange polynomial to construct
36 polynomial<T> L = {0};
37
38 for (int j = 0; j < points.size(); ++j) {
39
40 // The Lagrange polynomial is a linear
41 // combination of all l_j
42 polynomial<T> l_j = {1};
43
44 for (int m = 0; m < points.size(); ++m) {
45
46 if(m == j)
47 continue;
48
49 // l_j = product(x - x_m / x_j - x_m)
50 l_j *= polynomial<T>({-points[m].get(0), 1});
51 l_j /= points[j].get(0) - points[m].get(0);
52 }
53
54 // L = sum(y_j * l_j)
55 l_j *= points[j].get(1);
56 L += l_j;
57 }
58
59 return L;
60 }

67 inline vec_buff chebyshev_nodes(real a, real b, unsigned int n) {
68
69 vec_buff nodes;
70 nodes.resize(n);
71
72 real m = (b + a) / 2.0;
73 real c = (b - a) / 2.0;
74
75 for (int i = 1; i < n + 1; ++i)
76 nodes[i - 1] = m + c * cos(real(2 * i - 1) / real(2 * n) * PI);
77
78 return nodes;
79 }

89 inline polynomial<real> interpolate_grid(real_function f, real a, real b, unsigned int order) {
90
91 std::vector<vec2> points;
92 points.resize(order + 1);
93
94 // Sample <order + 1> equidistant points
95 for (int i = 0; i < order + 1; ++i) {
96 real x = (b - a) / real(order) * real(i);
97 points[i] = {x, f(x)};
98 }
