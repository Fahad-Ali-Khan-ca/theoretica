14 // Compute the Lagrange polynomial
15 // interpolating a set of points
16 template<typename T>
17 inline polynomial<T> lagrange_polynomial(const std::vector<vec<2, T>>& points) {
18
19 // Check that all x_i are different to prevent
20 // division by zero
21 for (int i = 0; i < points.size() - 1; ++i) {
22 if(points[i].get(0) == points[i + 1].get(0)) {
23 UMATH_ERROR("lagrange_polynomial", points[i].get(0), INVALID_ARGUMENT);
24 return polynomial<T>({T(nan())});
25 }
26 }
27
28 // Lagrange polynomial to construct
29 polynomial<T> L = {0};
30
31 for (int j = 0; j < points.size(); ++j) {
32
33 // The Lagrange polynomial is a linear
34 // combination of all l_j
35 polynomial<T> l_j = {1};
36
37 for (int m = 0; m < points.size(); ++m) {
38
39 if(m == j)
40 continue;
41
42 // l_j = product(x - x_m / x_j - x_m)
43 l_j *= polynomial<T>({-points[m].get(0), 1});
44 l_j /= points[j].get(0) - points[m].get(0);
45 }
46
47 // L = sum(y_j * l_j)
48 l_j *= points[j].get(1);
49 L += l_j;
50 }
51
52 return L;
53 }

56 // Compute the <n> Chebyshev nodes on a given interval
57 // from <a> to <b> (a < b)
58 inline vec_buff chebyshev_nodes(real a, real b, unsigned int n) {
59
60 vec_buff nodes;
61 nodes.resize(n);
62
63 real m = (b + a) / 2.0;
64 real c = (b - a) / 2.0;
65
66 for (int i = 1; i < n + 1; ++i)
67 nodes[i - 1] = m + c * cos(real(2 * i - 1) / real(2 * n) * PI);
68
69 return nodes;
70 }

73 // Compute the interpolating polynomial of a real function
74 // on an equidistant point sample.
75 // <f> is the function to interpolate
76 // <a> and <b> are the extremes of the interval (a < b)
77 // <order> is the order of the resulting polynomial
78 inline polynomial<real> interpolate_grid(real_function f, real a, real b, unsigned int order) {
79
80 std::vector<vec2> points;
81 points.resize(order + 1);
82
83 // Sample <order + 1> equidistant points
84 for (int i = 0; i < order + 1; ++i) {
85 real x = (b - a) / real(order) * real(i);
86 points[i] = {x, f(x)};
87 }
88
89 return lagrange_polynomial(points);
