62 template<typename Dataset>
63 histogram(const Dataset& data, unsigned int bin_count = 0) {
64
65 range_max = max(data);
66 range_min = min(data);
67 max_val = range_max;
68 min_val = range_min;
69 sum = theoretica::sum(data);
70 sqr_sum = sum_squares(data);
71 N = data.size();
72
73 // Default bin count is sqrt(N)
74 bins.resize(
75 bin_count ? bin_count : floor(sqrt(N))
76 );
77
78 // The histogram contains the data point
79 // by construction
80 for (size_t i = 0; i < N; ++i)
81 bins[index(data[i])]++;
82 }
