212 inline benchmark_result benchmark(const std::string& funcName, RealFunction f, RealInputGenerator g,
213 unsigned int n = state.defaultIterations, unsigned int m = state.defaultRuns) {
214
215 // Dummy variable
216 __volatile__ Real c = 0;
217
218 std::vector<Real> input;
219 input.reserve(n);
220
221 for (unsigned int i = 0; i < n; ++i)
222 input[i] = g(i);
223
224 // Sum of m runs with n iterations each
225 long double sum = 0;
226
227 for (unsigned int i = 0; i < m; ++i) {
228
229 timer t = timer();
230
231 for (unsigned int j = 0; j < n; ++j)
232 c += f(input[j]);
233
234 sum += t();
235 }
236
237 return benchmark_result(funcName, sum, n, m);
238 }

242 inline benchmark_result benchmark(
243 const std::string& funcName, RealFunction f,
244 const std::vector<Real> input,
245 unsigned int n = state.defaultIterations, unsigned int m = state.defaultRuns) {
246
247 if(input.size() < n) {
248 std::cout << "Wrong input size in benchmark, skipping request ..." << std::endl;
249 state.failedBenchmarks++;
250 return benchmark_result();
251 }
252
253 // Dummy variable
254 __volatile__ Real c = 0;
255
256 // Sum of m runs n iterations each
257 long double sum = 0;
258
259 for (unsigned int i = 0; i < m; ++i) {
260
261 timer t = timer();
262
263 for (unsigned int j = 0; j < n; ++j)
264 c += f(input[j]);
265
266 sum += t();
267 }
268
269 return benchmark_result(funcName, sum, n, m);
270 }

288 inline void run() {
289
290 std::cout.precision(8);
291
292 if(!state.quiet) {
293 std::cout << "Starting benchmark of " << state.moduleName << std::endl;
294 std::cout << "Parameters: Iterations = " << state.defaultIterations << ", Runs = " << state.defaultRuns << "\n" << std::endl;
295
296 std::cout << std::left << std::setw(20) << "Function" << " | "
297 << std::setw(12) << "Time (ms)" << " | " << std::setw(12) << "Runs/sec" << std::endl;
298 state.outputFile << "Function, Time(ms), Runs/sec" << std::endl;
299 }
300
301 for (const auto& r : state.requests) {
302
303 // Skip benchmark if it hasn't been picked
304 if(!state.pickedBenchmarks.empty() && !state.pickedBenchmarks[r.funcName])
305 continue;
306
307 benchmark_result br = benchmark(r);
308 state.results.push_back(br);
309
310 if(!state.quiet)
311 print_benchmark(br);
312
313 state.outputFile << br.funcName << ", "
314 << br.avg_time << ", "
315 << br.runs_per_sec << std::endl;
316 }
317
318 for (const auto& r : state.customRequests) {
319
320 // Skip benchmark if it hasn't been picked
321 if(!state.pickedBenchmarks.empty() && !state.pickedBenchmarks[r.funcName])
322 continue;
323
324 benchmark_result br = r.f(r.iter, r.runs);
325 br.funcName = r.funcName;
326
327 state.results.push_back(br);
328
329 if(!state.quiet)
330 print_benchmark(br);
331
332 state.outputFile << br.funcName << ", "
333 << br.avg_time << ", "
334 << br.runs_per_sec << std::endl;
335 }
336
337 state.requests.clear();
338 }

342 inline void terminate(bool exit = true) {
343
344 if(state.requests.size() + state.customRequests.size())
345 run();
346
347 std::cout << "\nFinished benchmark of " << state.moduleName << std::endl;
348 if(state.failedBenchmarks)
349 std::cout << state.failedBenchmarks << " benchmarks failed!" << std::endl;
350
351 std::string filename = state.outputFolder + "benchmark_" + state.moduleName + ".csv";
352 std::cout << "Results have been saved in " << filename << std::endl;
353
354 state.moduleName = "unknown";
355 state.outputFile.close();
356
357 if(exit)
358 std::exit(state.failedBenchmarks);
359 }
