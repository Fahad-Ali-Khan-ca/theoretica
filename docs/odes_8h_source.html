33 template<unsigned int M>
34 inline ode_state<N>& operator=(const std::array<real, M>& v) {
35
36 if(v.size() < 2) {
37 TH_MATH_ERROR("ode_state::operator=", v.size(), INVALID_ARGUMENT);
38 t = nan();
39 return *this;
40 }
41
42 t = v[0];
43 v.resize(v.size() - 1);

45 for (int i = 1; i < v.size(); ++i)
46 y[i - 1] = v[i];
47
48 return *this;
49 }
