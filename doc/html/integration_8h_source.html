62

127 // Runge-Kutta integration of 4th order
128 inline void integrate(kinematic_state s, real t, real dt, accel_function accel) {
129
130 kinematic_deriv a, b, c, d;
131
132 a = eval(s, t, 0, kinematic_deriv(), accel);
133 b = eval(s, t, dt * 0.5, a, accel);
134 c = eval(s, t, dt * 0.5, b, accel);
135 d = eval(s, t, dt, c, accel);
136
137 // Approximate integration using 4 different points
138 real dxdt = 1.0f / 6.0f * (a.dx + 2.0f * (b.dx + c.dx) + d.dx);
139 real dvdt = 1.0f / 6.0f * (a.dv + 2.0f * (b.dv + c.dv) + d.dv);
140
141 // Update state
142 s.x = s.x + dxdt * dt;
143 s.v = s.v + dvdt * dt;
144 }
